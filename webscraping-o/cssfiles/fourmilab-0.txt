/*

    /* Regular text links */
    
    a:link,  a:visited {
	background-color: inherit;

	text-decoration: none;
    }

    a:hover  {
    	background-color:  rgb(30%, 30%, 100%);
    	color: rgb(100%, 100%, 100%);
    }
    
    a:active {
    	color: rgb(100%, 0%, 0%);

	text-decoration: none;
