$(function(){
    $.ajax({
        url:'/newsrelease/atom.xml',
        type:'GET',
        dataType:'xml',
        timeout:1000,
        error:function() {
        //alert("ooops");
        },
        success: parse_xml
            
    });
});

function parse_xml(xml,status){
  if(status!='success')return;
  var count = 0;
  $(xml).find('entry').each(function() {
    count ++;
    var datetime = $(this).find('datetime').text();
    var url = $(this).find('url').text();
    var title = $(this).find('title').text();
    var tag= '';
    $(this).find('tag').each(function () {
       var tagClass = $(this).attr('class');
       var tagLabel = $(this).attr('label');
       tag = tag + '<span class="tag '+ tagClass +'">' + tagLabel + '</span>';
    });
     $("#newsFeed ul").append('<li><time datetime="'+datetime+'">'+datetime+'</time>' + tag + '<p class="news-title"><a href="' + url + '">' + title + '</a></p><span class="arrow-gray"><a href="' + url + '"><img src="/common/images/arrow_gray.png" height="36" width="36"></a></span></li>');
    if(count == 3) return false;
  });
}
