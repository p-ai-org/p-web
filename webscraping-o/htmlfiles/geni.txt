    function slideshow() {
  var container = $j('#slideshow');
  var quotes = container.find('li');
  quotes.hide().filter(':first').show();
  /* Create pagination */
  container.append('<ul id="pagination"></ul>');
  var pagination = $j('#pagination');
  for (i=1; i<=quotes.length; i++) {
    pagination.append('<li><a href="#slide' + i + '">' + i + '</a></li>');
  }
  pagination.find('li').filter(':first').addClass('first current').end().filter(':last').addClass('last');
  pagination.find('a').bind('click', function(){
    quotes.fadeOut().filter(this.hash).fadeIn();
    $j(this).parents('li').siblings().removeClass('current');
    $j(this).parents('li').addClass('current');
    return false;
  })
  /* Create next button */
  container.append('<a id="next" href="#next"></a>');
  var next = $j('#next');
  next.bind('click', function(){
    var current = pagination.find('li.current');
    if (current.hasClass('last')) {
      pagination.find('li.first a').click();
    } else {
      current.next().find('a').click();
    }
    return false;
  });
  /* Create previous button */
  container.append('<a id="prev" href="#prev"></a>');
  var prev = $j('#prev');
  prev.bind('click', function(){
    var current = pagination.find('li.current');
    if (current.hasClass('first')) {
      pagination.find('li.last a').click();
    } else {
      current.prev().find('a').click();
    }
    return false;
  })
  container.hover(
    function(){ stopRotation(); },
    function(){ startRotation(); }
  );
 }
 slideshow();

  if (startRotation != '') {

 }

   function activatePlaceholders() {
  if ('placeholder' in document.createElement('input')) return false;
  var inputs = document.getElementsByTagName("input");
  for (var i=0;i<inputs.length;i++) {
    if (inputs[i].getAttribute("type") == "text") {
      var placeholder = inputs[i].getAttribute("placeholder");
      if (placeholder && placeholder.length > 0) {
        inputs[i].value = placeholder;
        inputs[i].style.color = "#999";
        inputs[i].onfocus = function() {
          if (this.value == this.getAttribute("placeholder")) {
            this.style.color = "#000";
            this.value = "";
          }
          return false;
        }
        inputs[i].onblur = function() {
          if (this.value.length < 1) {
            this.style.color = "#999";
            this.value = this.getAttribute("placeholder");
          }
        }
      }
    }
  }
}

window.onload = function() {
  activatePlaceholders();
}

var isSubmitting = false;

function checkForm() {
  if (isSubmitting) return;
  var value_fname    = document.getElementById('fname').value;
  var value_lname    = document.getElementById('lname').value;
  var value_email    = document.getElementById('email').value;
  error_message.innerHTML = " ";
  var errorText =
    ( !value_fname || value_fname == 'First Name' ) ? "Please enter a first name"             :
    ( !value_lname || value_lname == 'Last Name'  ) ? "Please enter a last name"              :
    (  isValidEmail(value_fname)                  ) ? "First name cannot be an email address" :
    (  isValidEmail(value_lname)                  ) ? "Last name cannot be an email address"  :
    ( !isValidEmail(value_email)                  ) ? "Please enter a valid email"            : null;
  if(errorText) {
    error_message.show().update(tr(errorText, 'start page error message'));
    if(window._gaq) _gaq.push(["_trackEvent('start_page_v2', 'submit_button_error')"]);
    return;
  } else {
    submit_button.innerHTML = '<span>Please wait...</span>';
    isSubmitting = true;

    // check if flash is installed, if so, send them along to the HTML5 version of the tree.
    try{
      var flash_installed =
        (typeof navigator.plugins != "undefined" && typeof navigator.plugins["Shockwave Flash"] == "object") || (window.ActiveXObject && (new ActiveXObject("ShockwaveFlash.ShockwaveFlash")) != false);
    } catch(err)
    {
      var flash_installed = true;
    }

    if (!flash_installed)
      $j('#touch-device', '#start_form').prop('checked', true);

    document.forms["start_form"].submit();
    if(window._gaq) _gaq.push(["_trackEvent('start_page_v2', 'submit_button_success')"]);
  }
}
