  // Indicate scroll position in #navList
  var navListArray = [],docHeight;
  function navListCalculateScroll(){
     var windowPos = $(window).scrollTop();
     var headerHeight = $("div.desktop-header-container").height();
     var windowHeight = $(window).height();
     var divMargin = 30;
     var activateLine = windowPos + headerHeight + divMargin;
     for (var i=0; i < navListArray.length; i++) {
       var theID = navListArray[i];
       var divPos = $(theID).offset().top;
       var divHeight = $(theID).height();
       if ( activateLine >= divPos && activateLine < (divPos + divHeight)) {
         $("a").removeClass("nav-active");
         //$("a[href='" + theID + "']").addClass("nav-active");
         // line below is same as above but with fix for
         // the anchor/hashtag link problem
         $("a[href='" + document.location.pathname + theID + "']").addClass("nav-active");
       } //else {
         //$("a[href='" + theID + "']").removeClass("nav-active");
       //}
     }
    if(windowPos + windowHeight == docHeight) {
      if (!$("#navList li:last-child a").hasClass("nav-active")) {
        var navActiveCurrent = $(".nav-active").attr("href");
        //$("a[href='" + navActiveCurrent + "']").removeClass("nav-active");
        // line below is same as above but with fix for
        // the anchor/hashtag link problem
        $("a[href='" + document.location.pathname + navActiveCurrent + "']").removeClass("nav-active");
        $("#navList li:last-child a").addClass("nav-active");
      }
    }
  }

  function stickykitFloatingNav() {
    // Sticky-Kit / Make FloatingNav fixed inside parent
    var ww = Math.max(document.documentElement.clientWidth, window.innerWidth || 0);
    if(ww>=979){
      $("#secondarynav").stick_in_parent({offset_top: 90}).removeClass("under812").addClass("over812");
    } else{
      $("#secondarynav").trigger("sticky_kit:detach").removeClass("over812").addClass("under812");
    }
  }

  function addFloatNav(headline){
    // Adds floating navigation to #secondarynav, automatically detects and creates links for each section of the page
    var headertxt = "";
    $("#floatingNav").remove();
    if (!headline){ headertxt = "On This Page:";
    } else {        headertxt = headline;}
    var $div = "<div id='floatingNav' class='dropdown'><div class='dropdown-container'><h3 class='dropdown-description'>" + headertxt + "</h3><p class='dropdown-button'></p><ul id='navList' class='dropdown-menu dropdown-select'></ul></div></div>";
    var floatingLIs="",txt="",txtToID="";
    $("#container section").each(function() {
      if ( $(this).find("h2").first().text() !== "" && $(this).css("display") !== "none" && $(this).is(':visible') && $(this).outerHeight() > 0) {
        txt = $(this).find("h2").first().text();
        txtToID = txt.replace(/[^a-z0-9]/gmi, "-").replace(/\W+/g, "-").toLowerCase();
        if ( $(this).find("h2").first().data("localize") ){		
          var localize = " data-localize='"+ $(this).find("h2").first().data("localize") +"'";		
        } else {var localize = "";}
        if ( !$(this).attr('id') ) {
          //floatingLIs += "<li><a href='#" + txtToID + "'>" + txt+ "</a></li>"; // for use when <base href/> is not used
          floatingLIs += "<li><a"+localize+" href='"+ document.location.pathname +"#" + txtToID + "'>" + txt+ "</a></li>";// for use when <base href/> is used
          $(this).attr("id",txtToID);
        } else {
          //floatingLIs += "<li><a href='#" + $(this).attr('id') + "'>" + txt + "</a></li>"; // for use when <base href/> is not used
          floatingLIs += "<li><a"+localize+" href='"+ document.location.pathname +"#" + $(this).attr('id') + "'>" + txt + "</a></li>"; // for use when <base href/> is used
        }
      }
    });
    $("#secondarynav").append($div);
    $("#navList").append(floatingLIs);
    stickykitFloatingNav();
    var navListChildren = $("#navList li").children();
    for (var i=0; i < navListChildren.length; i++) {
        var aChild = navListChildren[i];
        //var ahref = $(aChild).attr('href'); // for use when <base href/> is not used
        var ahref = $(aChild).attr('href').split('#'); // for use when <base href/> is used
        var ahref = "#" + ahref[1];                    // for use when <base href/> is used
        navListArray.push(ahref);
    }
    navListCalculateScroll();
  }
