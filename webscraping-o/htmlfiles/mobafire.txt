         var ajaxPagedModuleIsBusy = false;

	$(document).on('click','._ajax-previous-page',function(){
		if( ajaxPagedModuleIsBusy ) return false;
		ajaxPagedModuleIsBusy = true;

		var oldDiv = $($(this).metadata().target);
		var page = $(oldDiv).metadata().page;
		var allowOverflow = $(this).metadata().overflow;
		var that = this;

		if( !page ) page = 1;
		if( !allowOverflow && page == 1 )
		{
			ajaxPagedModuleIsBusy = false;
			return false;
		}

		var url = '/ajax/module?page=' + (page-1);
		for( key in $(this).metadata() ) url += '&' + key + '=' + $(this).metadata()[ key ];

		var scroll = $(this).metadata().scroll;

		$.get(url,function(data){
			var newDiv = $(String(data));
			if( !allowOverflow ){
				$(newDiv).metadata().page = page-1;
			}

			if( scroll == 'none' )
			{
				$(oldDiv).before($(newDiv));
				$(oldDiv).remove();
				ajaxPagedModuleIsBusy = false;
			}
			else if( scroll == 'horizontal' )
			{
				var width = $(oldDiv).width();

				//shift _viewport left to make room
				$('._viewport', $(that).parent() ).css('left','-'+width+'px');
				//add the new div before the old
				$(oldDiv).before($(newDiv));
				//animate _viewport to the right
				$('._viewport', $(that).parent()).animate({left: '0px'},500,function(){
					$(oldDiv).remove();
					ajaxPagedModuleIsBusy = false;
				});
				/*

				$(newDiv).animate({marginTop: 0},500,function(){
					$(oldDiv).remove();
					ajaxPagedModuleIsBusy = false;
				});
			}
		});
	});

				$('._viewport', $(that).parent()).css('left','0px');
				//add in new div
