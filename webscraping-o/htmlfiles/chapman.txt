  <script>
   var contactEl;
        var toggleBtn;
        var contactElHidden = true;

        $(function () {
            contactEl = $('#covid19-callout-box');
            toggleBtn = contactEl.find('.covid19-btn-opener');
            toggleBtn.click(_toggleContactEl);
            $(window).scroll(_onScroll);
        });

        function _toggleContactEl() {
            TweenLite.to(contactEl, 0.3, {
                x: contactElHidden ? '0%' : '100%'
            });
            contactElHidden = !contactElHidden;
            $("#covid-carot i").toggleClass('fa-caret-up fa-caret-down');
            $("#covid19-learn-more-btn").attr('tabindex', function(index, attr){
                return attr == 0 ? -1 : 0;
            });
        }

        function _onScroll() {
            // scrolled half-way
            if ($(window).scrollTop() < $(document).height() * 0.5) {
                if (contactElHidden) {
                    _toggleContactEl();
                }
                // only try to display once
                $(window).off('scroll');
            }
        }
  </script>
