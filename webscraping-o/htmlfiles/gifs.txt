function checkForUrlParam(urlParam) {
  var detected = false;
  queryString = document.location.search;
  if (queryString.length <= 3) {
    return null;
  }
  source = null;
  params = queryString.substr(1).split('&');
  params.forEach(function(param) {
    p = param.split(/\=(.+)?/);
    if (p[0] === urlParam) {
      detected = true;
    }
  });
  return detected;
}
