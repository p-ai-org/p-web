  <script crossorigin="anonymous" integrity="sha256-CSXorXvZcTkaix6Yvo6HppcZGetbYMGWSFlBw8HfCJo=" src="https://code.jquery.com/jquery-3.4.1.min.js">

  <script src="https://cdnjs.cloudflare.com/ajax/libs/dropzone/5.4.0/dropzone.js">

  <script type="text/javascript">
   Dropzone.autoDiscover = false;
    var csrf_token = $('meta[name="csrf-token"]').attr('content');
    jQuery.ajaxSetup({
        headers: {
            'X-CSRF-TOKEN': csrf_token
        }
    });
    (function ($) {
        $('[data-confirm]').click(function () {
            var msg = $(this).attr('data-confirm');
            if(!window.confirm(msg?msg:'Are you sure you want to perform this action?')){
                return false;
            }
        });
        $('[data-toggle="full"]').click(function(){
            $('.view-container').toggleClass('full');
            return false;
        });

        var config = {
            sessionEndpoint: 'https://www.imagevenue.com/upload/session',
            uploadEndpoint: 'https://www.imagevenue.com/upload',
            completeEndpoint: 'https://www.imagevenue.com/upload/complete',
            maximumFiles: 500,
            parallelUploads: 5,
            generalError: 'There was an error uploading your files. Please try again later.'
        };
        $('[data-uploader]').each(function () {
            var $this = $(this);
            var $dropArea = $this.find("[data-uploader-drop-area]");
            var $options = $this.find("[data-uploader-options]");
            var $galleryTitle = $this.find("[data-uploader-gallery-option-value-create]");
            var $gallerySelect = $this.find("[data-uploader-gallery-option-value-select]");
            var $comments = $this.find("[data-uploader-comments-option]");
            var $contentType = $this.find("[data-uploader-content-type]");
            var $thumbnailSize = $this.find("[data-uploader-thumbnail-option]");
            var $galleryOption = $this.find('[data-uploader-gallery-option]');
            var $start = $this.find('[data-uploader-start]');
            var $errorPlaceholder = $this.find('[data-uploader-error]');
            var $successPlaceholder = $this.find('[data-uploader-success]');
            var $preview = $this.find('[data-uploader-preview]');
            var $progress = $this.find('[data-uploader-progress]');
            var $progressBar = $this.find('[data-uploader-progressbar]');
            var $output = $this.find('[data-uploader-out]');

            var token = null;
            var session = null;
            $galleryOption.change(function () {
                if ($(this).is(':checked')) {
                    $galleryTitle.show();
                    $gallerySelect.hide();
                } else {
                    $galleryTitle.hide();
                    $gallerySelect.show();
                }
            });

            var getData = function () {
                var data = {
                    "thumbnail_size": $thumbnailSize.val(),
                    "content_type": $contentType.val(),
                    "comments_enabled": $comments.val() === "1",
                };
                if ($galleryOption.is(':checked')) {
                    data.gallery = true;
                    data.gallery_title = $galleryTitle.val();
                }
                if ($gallerySelect.val()) {
                    data.gallery_token = $gallerySelect.val()
                }
                return data;
            };
            var reset = function () {
                $progress.hide();
                $progressBar.text("");
                $progressBar.attr('style', 'width:0%');
                $preview.html("");
                $options.hide();

            };
            var onProgressChange = function (progress) {
                $progressBar.attr('style', 'width:' + progress + '%');
                $progressBar.text(parseInt(progress) + "%");
            };
            var getToken = function (error, success) {
                $.ajax({
                    method: "POST",
                    url: config.sessionEndpoint,
                    data: getData(),
                    dataType: 'json'
                }).fail(function (er) {
                    onError(config.generalError, true);
                }).done(function (da, textStatus, jqXHR) {
                    if (da.error) {
                        onError(config.generalError, true);
                    } else {
                        token = da.data;
                        session = da.session;
                        success();
                    }
                });
            };
            var onComplete = function (files) {
                $errorPlaceholder.hide();
                $successPlaceholder.html("Your files have been uploaded!");
                $successPlaceholder.show();
                window.location.href = config.completeEndpoint + '?session=' + session;

            };
            var onError = function (message, error) {
                $successPlaceholder.hide();
                $errorPlaceholder.show();
                $errorPlaceholder.text(message);

            };
            var uploadedFiles = [];
            var $dropZone = $this.dropzone({
                url: config.uploadEndpoint,
                withCredentials: true,
                uploadMultiple: true,
                clickable: $dropArea[0],
                timeout: 180000,
                maxFiles: config.maximumFiles,
                parallelUploads: config.parallelUploads,//config.maximumFiles,
                maxFilesize: 1024 * 10,
                acceptedFiles: ".jpg,.png,.gif,.jpeg",
                createImageThumbnails: true,
                previewTemplate: '<div class="preview-image"><a href="javascript:void(0);" data-uploader-remove-item=""><i class="far fa-trash-alt"></i></a><img data-dz-thumbnail/></div>',
                previewsContainer: $preview[0],
                autoProcessQueue: false,
                paramName: "files",
                init: function () {
                    var $zone = this;
                    this.on('addedfile', function (file) {
                        $(file.previewElement).find('[data-uploader-remove-item]').click(function () {
                            $zone.removeFile(file);
                        });
                        $options.show();
                    });
                    this.on('removedfile', function (file) {
                        if (this.getQueuedFiles().length === 0) {
                            reset();
                        }
                    });
                },
                totaluploadprogress: function (progress) {
                    onProgressChange(progress);
                },
                sending: function (file, xhr, formData) {
                    formData.append("_token", csrf_token);
                    xhr.ontimeout = function (e) {
                        onError('Uploading ' + file.name + ' timed out.');
                    };
                },
                params: function () {
                    return {'data': token};
                },
                successmultiple: function (file, response) {
                    try {
                        uploadedFiles.push(file);
                        $dropZone[0].dropzone.processQueue();
                    } catch (ex) {
                        onError(config.generalError, true);
                    }
                },
                queuecomplete: function (files, response) {
                    $start.show();
                    try {
                        onComplete(uploadedFiles, response);
                    } catch (ex) {
                        onError(config.generalError, true);
                    }
                },
                complete: function () {

                },
                error: function (file, message, xhr) {
                    if (!file.accepted) {
                        this.removeFile(file);
                    }
                    if (xhr) {
                        onError(config.generalError, true);
                    } else {
                        onError(message);
                    }
                },
                maxfilesexceeded: function () {
                    onError("You can't upload more than " + config.maximumFiles + " files");
                }
            });
            $start.click(function () {
                var d = getData();
                if (d.gallery && d.gallery_title.trim().length == 0) {
                    alert("Please enter a name for your gallery");
                    return false;
                }
                if ($dropZone[0].dropzone.getQueuedFiles().length > 0) {
                    try {
                        if (typeof ga !== 'undefined') {
                            ga('send', 'event', 'widget', 'imgbox', 'upload', $dropZone[0].dropzone.getQueuedFiles().length);
                        } else if (typeof _gaq !== 'undefined') {
                            _gaq.push(['_trackEvent', 'widget', 'imgbox', 'upload', $dropZone[0].dropzone.getQueuedFiles().length]);
                        }
                    } catch (err) {
                    }
                    $start.hide();
                    onProgressChange(0);
                    $progress.show();

                    getToken(function (error) {
                        onError(config.generalError, true);
                    }, function (success) {
                        $dropZone[0].dropzone.processQueue();
                    });
                } else {
                    alert("You need to select some files first!");
                }
            });
        });
    })(jQuery);
  </script>
