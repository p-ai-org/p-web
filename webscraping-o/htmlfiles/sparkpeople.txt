   function readCookie(name) {
	var nameEQ = name + "=";
	var ca = document.cookie.split(';');
	for(var i=0;i < ca.length;i++) {
		var c = ca[i];
		while (c.charAt(0)==' ') c = c.substring(1,c.length);
		if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
	}
	return null;
}

function findPosY(obj) {
	var curtop = 0;
	if (typeof (obj.offsetParent) != 'undefined' && obj.offsetParent) {
		while (obj.offsetParent) {
			curtop += obj.offsetTop;
			obj = obj.offsetParent;
		}
		curtop += obj.offsetTop;
	}
	else if (obj.y)
		curtop += obj.y;
	return curtop;
}

function createCookie(name,value,days) {
	if (days) {
		var date = new Date();
		date.setTime(date.getTime()+(days*24*60*60*1000));
		var expires = "; expires="+date.toGMTString();
	}
	else var expires = "";
	document.cookie = name+"="+value+expires+"; path=/";
}

function readCookie(name) {
	var nameEQ = name + "=";
	var ca = document.cookie.split(';');
	for(var i=0;i < ca.length;i++) {
		var c = ca[i];
		while (c.charAt(0)==' ') c = c.substring(1,c.length);
		if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
	}
	return null;
}

      {
      Char = sText.charAt(i);
      if (ValidChars.indexOf(Char) == -1)
         {
         IsNumber = false;
         }
      }
   return IsNumber;

function OldEnough(BirthDay)
{
	var dateToday = new Date();
	var monthEnd = BirthDay.indexOf("/");
	var dayStart = BirthDay.indexOf("/") + 1;
	var dayEnd = BirthDay.indexOf("/",dayStart);
	var yearStart = BirthDay.indexOf("/",dayEnd) + 1;
	
	var year = BirthDay.substring(yearStart);

	if ((dateToday.getFullYear() - year) <= 18)
	{
		if ((dateToday.getFullYear() - year) < 18)
		{
			return (false);
		}
		
		var month = BirthDay.substring(0,monthEnd);
		
		if (dateToday.getMonth() + 1 <= month)
		{
			if (dateToday.getMonth() + 1 < month)
			{
				return (false);
			}
			
			var day = BirthDay.substring(dayStart,dayEnd);

			

			if (dateToday.getDate() < day)
			{
				return(false);
			}
		}
	}

	return (true);
}

		return (false);

		return (false);

		alert("Your username can not contain the # character. Please enter a new value for the UserName field.");

		return (false);

		return (false);

		return (false);

		return (false);

		alert("Please no apostrophes in the password. Please enter a new value for the \"Password\" field.");    
		theForm.strPassword.focus();

function bOnlyValidChars(strParam)
{
	var strValidChars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789_-';
	for(i = 0; i < strParam.length; i++)
	{
		if (strValidChars.indexOf(strParam.charAt(i)) < 0)
		{
			return false;
		}
	}
	return true;
}
