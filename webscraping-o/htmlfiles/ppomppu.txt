    function isAdDivVisible( obj )
    {
        if( obj.length == 0 ) return false;
        var target           = obj.offset().top;      // 
        var window_height    = $(window).height();    //
        var window_scrollTop = $(window).scrollTop(); //  
        var gap              = 500;                    // px
        var isVisible        = false;

        isVisible = ( target - gap <= (window_height) )                    ? true : isVisible;

        return isVisible;

                };
            }else{
                return false;
            }
            break;

        case 'twitter' :
            if( url.match(/\/status\//) ){
                api_url = "https://publish.twitter.com/oembed?callback=?";
                data = {"url" : url};
            }else{
                return false;
            }
            break;

        case 'facebook' :
            // video

            }else if( url.match(/\/posts\/|\/activity\/|\/photos\/|\/media\/|\/questions\/|\/notes\/|\/photo.php\?|\/permalink.php/) ){

                api_url = "https://graph.facebook.com/v9.0/oembed_page?";
