   <script>
    (function () {
        $(document).ready(function () {

            const $_searchbar = $('#search_bar');
            $('[data-toggle="search_bar"]').on('click', function () {
                if ($_searchbar.hasClass('d-none')) {
                    $_searchbar.removeClass('d-none');
                    $('#auto_complete').trigger('focus');
                } else {
                    $_searchbar.addClass('d-none');

                }


            })

            // Submit on button click behavior
            $('#auto_complete_button').click(function (e) {
                e.preventDefault();
                if (searchText) {
                    savedUserSearch(searchText, {});
                }
            });
            var searchText;
            var $_web_input_container = $('#web_input_container')
            $_web_input_container.on('input', function (e) {
                searchText = sanitizer.sanitize(e.target.value);
            });
            var xhr;
            var nbResults = 0;
            var url = '/en/artworks?q=term';
            var input_web = document.getElementById("auto_complete");


                                    if (input_web) autoCompleter(autoCompleterOption(input_web, 'results-web', $_web_input_container));


            /**
             * @see https://github.com/kraaden/autocomplete
             * AutoCompleter Option
             * @Param input input tag
             * @Param className autocomplete render block class name
             * @Param resultContainer jquery element that will contain the results
             * return Object Option for autoCompleter
             */
            function autoCompleterOption(input, className, resultContainer) {
                return {
                    input: input,
                    className: className,
                    debounceWaitMs: 200,
                    fetch:
                        function (text, update) {
                            text = sanitizer.sanitize(text).trim();
                            if (!text) {
                                return
                            }
                            /*
                            * To display "Loading..."
                            * pending the return of the request
                            * If the user submits the form before the response, it will be redirected
                            * to search_artwork route
                            */
                            var loading = {loading: true};
                            update([loading]);
                            update(false);
                            var suggestions;

                            if (xhr != null) {
                                xhr.abort();
                                xhr = null;
                            }
                            xhr = $.ajax({
                                type: 'GET',
                                url: '/en/esearch.json',
                                data: {q: text}
                            }).done(function (data) {
                                nbResults = data.totalFound;
                                suggestions = [{
                                    seeFullResult: data.seeFullResult
                                }];
                                if (data.results) {
                                    data.results.forEach(function (element) {
                                        suggestions.push(element)
                                    })
                                }
                                update(suggestions);
                            });
                        },
                    // Render Templating
                    render: function (suggestion, currentInput) {
                        var content;
                        var reg = new RegExp(currentInput, 'gi');
                        var itemElement = document.createElement("div");
                        itemElement.classList.add('suggestion_render');

                        if (suggestion.loading) {
                            content = '<div class="suggestion_item p-2"><p class="my-3" >Loading...</p></div>';
                            itemElement.innerHTML = content;
                            return itemElement;
                        }


                        if (suggestion.seeFullResult) {
                            content = '<div class="suggestion_item p-2"><p class="my-3">See full results for {s}</p></div>';
                            content = content.replace('{s}', '<span class="font-weight-bold">« ' + sanitizer.sanitize(input.value) + ' »</span>');
                        } else {
                            var cleanText = sanitizer.sanitize(suggestion.text)
                            var boldText = cleanText.replace(reg, function (optionText) {
                                return '<span class="text-dark font-weight-bold">' + optionText + '</span>'
                            });
                            var subText = sanitizer.sanitize(suggestion.subText);
                            subText = subText.replace(suggestion.artistName, function (optionText) {
                                return optionText.replace(reg, function (boldText) {
                                    return '<span class="text-dark s font-weight-bold">' + boldText + '</span>'
                                })
                            });

                            var cssClass = suggestion.type === 'artist' ? 'img-circle border' : '';
                            content = '<div class="d-flex p-2 suggestion_item ' + suggestion.type + '">' +
                                '<img class="' + cssClass + '" style="width: 45px; height: 45px;" src="' + suggestion.imgUrl + '" alt="' + cleanText + '">' +
                                '<div class="ml-2 d-flex flex-column">' +
                                '<span class="text-result">' + boldText + '</span>' +
                                '<span class="h6 text-result text-secondary">' + subText + '</span>' +
                                '</div>' +
                                '</div>';
                        }

                        itemElement.innerHTML = content;
                        return itemElement;
                    },
                    onSelect: function (element, autocomplete) {
                        savedUserSearch(sanitizer.sanitize(autocomplete.value), element);
                    },
                    emptyMsg: 'No results found.',
                    preventSubmit: false,
                    showOnFocus: true,
                    // Required to set the final position of the results
                    // By default result tag is applied at the bottom of the document with fixed style
                    // when we want it in the defined tag ( Styles are also customized )
                    customize: function (input, inputRect, container, maxHeight) {
                        resultContainer.append($(container))
                    }
                };


            }

            function savedUserSearch(string, element) {
                if (!element.url) {
                    element.url = url.replace("term", string)
                }
                $.ajax({
                    type: 'POST',
                    url: '/en/saved-user-search',
                    data: {string: string, nbResults: nbResults}
                }).done(function (data) {
                        window.location.href = element.url
                    }
                ).fail(function () {
                    window.location.href = element.url
                });

            }
        })
    })();

                if (entry.isIntersecting) {
                                        if (entry.intersectionRatio >= 0.75) {
                                                $_element.data('visible', true)

                                                        if ($_element.data('visible')) {

                function createFormData(obj, form, namespace) {
            let fd = form || new FormData();
            let formKey;

            for (let property in obj) {
                if (obj.hasOwnProperty(property) && obj[property]) {
                    if (namespace) {
                        formKey = namespace + '[' + property + ']';
                    } else {
                        formKey = property;
                    }
                    // if the property is an object, but not a File, use recursivity.
                    if (typeof obj[property] === 'object' && !(obj[property] instanceof File)) {
                        createFormData(obj[property], fd, formKey);
                    } else { // if it's a string or a File object
                        fd.append(formKey, obj[property]);
                    }
                }
            }

            return fd;
        }
