      jQuery(this.wrap).removeClass('hidden');
    };
    SliderBestMC.prototype.prepareHtml = function (postsArr) {
      var html = '';
      postsArr.forEach(function (item, i, arr) {
        var title_html,
            divide_word = ' in ',
            dividePos = item.title.indexOf(divide_word) + divide_word.length;
        title_html = '<span class="f_row_title">' + item.title.substring(0,dividePos) + '</span>'
                   + ' <span class="s_row_title">' + item.title.substring(dividePos) + '</span>';
        html += '<a href="'+ item.url +'">'+ title_html +'</a>';
      });
      this.putData(html);
    }
    SliderBestMC.prototype.putData = function(html){
      this.position = this.centerPosition;
      this.moveSlider(this.position);
      jQuery(this.slider).append(html);
      this.initCarousel();
      this.clickHandler();
    };
    SliderBestMC.prototype.initCarousel = function () {
      var self = this;
      self.timerId = setTimeout(function(){
        self.sliderCircle();
        self.initCarousel();
      }, self.delay);
    };
    
    SliderBestMC.prototype.sliderCircle = function(direction, isClick){
      isClick = isClick || false;
      direction = direction || 'left';
      var self = this;
      if(direction === 'left'){
        jQuery(self.slider).find('a:first').appendTo(jQuery(self.slider));
        self.position = self.centerPosition+100;
        self.moveSlider(self.position);
        self.position -= 100;

        setTimeout(function(){
          self.moveSliderSmooth(self.position, isClick ? self.animSpeedClick : self.animSpeed);
        },100);
      }else{
          jQuery(self.slider).find('a:last').prependTo(jQuery(self.slider));
          self.position = self.centerPosition-100;
          self.moveSlider(self.position);
          self.position += 100;
          setTimeout(function(){
            self.moveSliderSmooth(self.position, isClick ? self.animSpeedClick : self.animSpeed);
          },100);
      }
    };
    SliderBestMC.prototype.moveSliderSmooth = function(position, speed){
      jQuery(this.slider).css({
        '-webkit-transition': 'left '+speed+' cubic-bezier(0.39, 0.58, 0.57, 1)',
           '-moz-transition': 'left '+speed+' cubic-bezier(0.39, 0.58, 0.57, 1)',
                'transition': 'left '+speed+' cubic-bezier(0.39, 0.58, 0.57, 1)',
                      'left': position+'%'
      });
    };
    SliderBestMC.prototype.moveSlider = function(position){
      jQuery(this.slider).css({
        '-webkit-transition': 'none',
           '-moz-transition': 'none',
                'transition': 'none',
                      'left': position+'%'
      });
    };
    SliderBestMC.prototype.clickHandler = function(){
      var self = this;
      jQuery(self.arrows).click(function(){
        clearTimeout(self.timerId);
        var isLeft = jQuery(this).data().arrow_back;
        isLeft ? self.sliderCircle('right', true) : self.sliderCircle('left', true);
        self.timerId = setTimeout(function(){
          self.initCarousel();
        },self.delay);
      });
    };
    
});
