										delCoordinatorMainHTML += "<br>"+ delgateArr[index_del].Salutation + " " + delgateArr[index_del].FirstName + " " + delgateArr[index_del].LastName + " <a href='mailto:"+delgateArr[index_del].email+"' class='blue'>" + delgateArr[index_del].email+"</a> ("+delgateArr[index_del].Type+")";
									}
									else{
										delCoordinatorAlterHTML += "<br>"+ delgateArr[index_del].Salutation + " " + delgateArr[index_del].FirstName + " " + delgateArr[index_del].LastName + " <a href='mailto:"+delgateArr[index_del].email+"' class='blue'>" + delgateArr[index_del].email+"</a> ("+delgateArr[index_del].Type+")";
									}
									//countryName = delgateArr[index_del].Country;
									countryCode = delgateArr[index_del].DOL_GROUP;
									countryCode = countryCode.substring(4, 7);
								});
								delCoordinatorHTML = delCoordinatorMainHTML + delCoordinatorAlterHTML;
								$("#delCord").html(delCoordinatorHTML);
								$("#acc_country").html(getCountryArticle(countryCode));
								$("#popup-click").trigger('click');
							}
							
						}
					}
				});
				
			}
		}
	}   
}	
function getCountryArticle(countryCode){
	var returnCountryArticle = "";
	if(countryCode == "TPK"){
		countryCode = "CHT";
	}
	if(countryCode == "EUC"){
		countryCode = "EEC";
	}
	for(var cc=0; cc < country.length; cc++){
		if(country[cc].code == countryCode){
			returnCountryArticle = "<b>" + country[cc].name + "</b>";
			break;
		}
	}
	return returnCountryArticle;
}
