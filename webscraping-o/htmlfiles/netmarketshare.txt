    function restoreCustomState() {
        if (report && report.state) {
            //  restore custom state
            $('#dateInterval').val(report.state.dateInterval);
            $('#daySelect').css('display', 'none');
            $('#weekSelect').css('display', 'none');
            $('#monthSelect').css('display', 'none');

            switch (report.state.dateInterval) {
                case 'Daily':
                    report.settings.entity = 'dailyShare';
                    $('#dayStart').val(report.state.dateStart);
                    $('#dayEnd').val(report.state.dateEnd);
                    $('#daySelect').css('display', 'inline-table');
                    break;
                case 'Weekly':
                    report.settings.entity = 'weeklyShare';
                    $('#weekStart').val(report.state.dateStart);
                    $('#weekEnd').val(report.state.dateEnd);
                    $('#weekSelect').css('display', 'inline-table');
                    break;
                default:
                    if (report.settings.entity != 'monthlyHTTPS' && report.settings.entity != 'monthlyOrganization')
                        report.settings.entity = 'monthlyShare';

                    $('#dateInterval').val('Monthly');

                    if (report.state.dateStart)
                        $('#monthStart').val(report.state.dateStart);
                    else
                        $('#monthStart').val(report.settings.ranges[0].Trend[0]);

                    if (report.state.dateEnd)
                        $('#monthEnd').val(report.state.dateEnd);

                    $('#monthSelect').css('display', 'inline-table');
                    break;
            }
        }
        $('#custom-toolbar').css('display', 'inline-table');
    }
