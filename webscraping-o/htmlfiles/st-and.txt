      <script>
       window.onload = function() {
       
        var timeout = null;

         $("#form-banner-search").submit(function(e){
                e.preventDefault();
            });

        $("#form-banner-search").keyup(function() {
           
            if ( $(this).val().length > 2 ) {
                //AJAX search course
                var str = $(this).val().replace(/[0-9`~!@#$%^&*()_|+\-=?;:'",.<>\{\}\[\]\\\/]/gi,'');
                searchCourses(str);
            } 
            else {
                $(".results").slideUp('slow');
                $(".results .col-md-6.left-column").text("");
                $(".results .col-md-6.right-column").text("");
            }            

            //send search results to Google Analytics
            if ( $(this).val().length > 4 ) {

               gaquery = $(this).val().toLowerCase();

               if (timeout !== null) {
                   clearTimeout(timeout);
               }

               timeout = setTimeout(function() {
                  ga('send', 'event', 'Course search term', 'submit', gaquery);
               }, 500);
            
            }

        });
    };   

    function searchCourses(val){

      var info = val;

      $.ajax({
          type:'POST',
          url:'https://www.st-andrews.ac.uk/php/homepage/search_courses.php',
          data: {info},
          success: function(data){
            if(data !=''){
              $('#course_results').html(data);
              $(".results").slideDown('slow');              
            }
            else {
                $(".results").slideUp('slow');
                $(".results .col-md-6.left-column").text("");
                $(".results .col-md-6.right-column").text("");
            }
          }
      });
    }
      </script>
