		_tonum: function(str) {
			var hash = 0, i, chr;
			str = str + '';
			if (str.length === 0) return hash;
			for (i = 0; i < str.length; i++) {
				chr   = str.charCodeAt(i);
			    hash  = ((hash << 5) - hash) + chr;
			    hash |= 0; // Convert to 32bit integer
			}
			hash = hash >>> 0;
			return hash;
		},
