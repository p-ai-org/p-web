  <script>
   var Webflow = Webflow || [];
Webflow.push(function() {
    var heroShapeAni = $(".hero-shape-animation");
    var isHeroShapeAniPlayed = false;

    // Update the desktop navbar bg on scroll
    window.addEventListener('scroll', () => {
        let navScrollHeightPercent = document.documentElement.scrollHeight * 0.02;
        let currentPOS = document.documentElement.scrollTop || document.body.scrollTop;

        // Once the scroll height has gone past the % stated
        if (currentPOS >= navScrollHeightPercent) {
            $( ".navbar" ).css("background-color", "#ffffff");
        };

        // If the page is scrolled back to top
        if (currentPOS <= navScrollHeightPercent) {
            $( ".navbar" ).css("background-color", "transparent");
        };

        // Check if the hero shape animation is on the page
        if (heroShapeAni.length) {
            let heroScrollHeightPercent = document.documentElement.scrollHeight * 0.06;
            
            // Once the scroll height has gone past the % stated, trigger the rest of the animation
            if (currentPOS >= heroScrollHeightPercent && isHeroShapeAniPlayed == false) {
                isHeroShapeAniPlayed = true;
                $( "#hero-animation-play" ).trigger( "click" );
            };

            // If the page is scrolled back to top
            if (currentPOS <= heroScrollHeightPercent && isHeroShapeAniPlayed == true) {
                isHeroShapeAniPlayed = false;
                $( "#hero-animation-rewind" ).trigger( "click" );
            };
        };

    });
});
