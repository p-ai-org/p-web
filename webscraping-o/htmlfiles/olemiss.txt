function makeSearch() {
	var q = document.getElementById('q');
	var c = document.getElementById('search-category');
	var category = 'all';
	if (c.selectedIndex > -1) {
		category = c.options[c.selectedIndex].value;
	}
	var url = 'https://olemiss.edu/search/' + category + '/' + q.value;
	window.location.href = url;
	return false;
}

/* Equal Heights Plugin */
(function($) {
	$.fn.equalHeights = function(minHeight, maxHeight) {
		tallest = (minHeight) ? minHeight : 0;
		this.each(function() {
			if($(this).height() > tallest) {
				tallest = $(this).height();
			}
		});
		if((maxHeight) && tallest > maxHeight) tallest = maxHeight;
		return this.each(function() {
			$(this).height(tallest).css("overflow","auto");
		});
	}
})(jQuery);

// AutoScroll Announcements
	var period     = 6000;
	var transition = 600;
	function scrollUp() {
		getDeviceWidth();
		if (devicewidth >= pagesizes[2]) {
			var itemHeight  = $('#announcements li').height();
			var totalHeight = $('#announcements li').size() * itemHeight;
			var currentTop  = parseInt($('#announcements li').css('top'), 10);
			var newTop      = currentTop - itemHeight;
			if (-newTop < totalHeight) {
				$('#announcements li').animate({top: newTop});
			} else {
				newTop = 0;
				$('#announcements li').css({top: itemHeight}).animate({top: newTop});
			}
		} else {
			$('#announcements li').css({top: 0});
		}
	}
	setInterval(function() {scrollUp()}, period);
	
// Fix column heights for #newsannouncements and #eventphototweet
