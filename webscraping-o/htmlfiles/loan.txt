function submitFrmRates(FHALoanLimit, VALoanLimit){
	var fel = document.getElementById("frmRatesHp").elements;
	var loanAmt = parseInt(fel["USER_LOAN_AMOUNT"].value.replace(intValRE, ""));

	var validated = true;
	if(fel['USER_STATE'].selectedIndex == 0){
		document.getElementById('errorState').style.marginTop = '-8px';
		document.getElementById('errorState').innerHTML = 'Please select a state.';
		validated = false;
	} else if(fel['USER_STATE'].selectedIndex > 0) {
		document.getElementById('errorState').innerHTML = '';
		document.getElementById('errorState').style.marginTop = '0';
	}
	if(fel['USER_LOAN_PROGRAM'].selectedIndex == 0){
		document.getElementById('errorLoanProgram').style.marginTop = '-8px';
		document.getElementById('errorLoanProgram').innerHTML = 'Please select a loan type.';
		validated = false;
	} else if(fel['USER_LOAN_PROGRAM'].selectedIndex > 0) {
		document.getElementById('errorLoanProgram').innerHTML = '';
		document.getElementById('errorLoanProgram').style.marginTop = '0';
	}
	if(isNaN(loanAmt) || loanAmt <= 0){
		document.getElementById('errorLoanAmount').innerHTML = 'Please enter a valid loan amount.';
		validated = false;
	} else {
		document.getElementById('errorLoanAmount').innerHTML = '';
	}

	var loan_purpose;
	if (fel['USER_LOAN_PURPOSE'].value == 1)
	{
		loan_purpose = 'purch';
	}
	else if (fel['USER_LOAN_PURPOSE'].value == 3)
	{
		loan_purpose = 'equity';
	}
	else
	{
		loan_purpose = 'refi';
	}

	var href = fel['USER_STATE'].value+"/"+loan_purpose+"/"+fel['USER_LOAN_PROGRAM'].value+"/"+fel['USER_POINTS'].value+"/"+loanAmt;

	if (fel["USER_LOAN_PROGRAM"].value.indexOf("HE") >= 0) {
		if(loanAmt < 10000){
			document.getElementById('errorLoanAmount').innerHTML = 'Home Equity loan amount must be at least $10,000.';
			validated = false;
		}
	} else {
		if (loanAmt < 10000) {
			document.getElementById('errorLoanAmount').innerHTML = 'Mortgage loan amount must be at least $10,000.';
			validated = false;
		}
	}
	if (fel["USER_LOAN_PROGRAM"].options[fel["USER_LOAN_PROGRAM"].selectedIndex].text.indexOf("FHA") >= 0) {
		if (loanAmt > FHALoanLimit) {
			document.getElementById('errorLoanAmount').innerHTML = 'Maximum FHA Loan Amount is $' + FHALoanLimit;
			validated = false;
		}
	} else if (fel["USER_LOAN_PROGRAM"].options[fel["USER_LOAN_PROGRAM"].selectedIndex].text.indexOf("VA") >= 0) {
		if (loanAmt > VALoanLimit) {
			document.getElementById('errorLoanAmount').innerHTML = 'Maximum VA Loan Amount is $' + VALoanLimit;
			validated = false;
		}
	}

	if (!validated)
	{
		return validated;
	}

	href = window.location.protocol +"//"+ window.location.host +"/mortgage-rates/"+href;
	window.location.href = href;
}
