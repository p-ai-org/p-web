            function updateHiddenInputsAndAction(formID_) {

                                                formID = "#" + formID_;

                                                var selectedText = $('button[data-sitecore-js-target]').text();

                                                // Remove existing hidden fields
                                                $('input[data-funnelback-hidden="hidden"]').remove();

                                                // Add the selected option's hidden fields
                                                var selectedSearchOptions = {"collection":"uwa-meta","form":"uwa","profile":"_default","scope":"uwa.edu.au"};

                                                if (selectedText == "Search All UWA") {
                                                    selectedSearchOptions = {"collection":"uwa-meta","form":"uwa","profile":"_default"};
                                                } 
                                                
                                                for (var key in selectedSearchOptions) {
                                                    $('<input>').attr({
                                                        type: 'hidden',
                                                        name: key,
                                                        value: selectedSearchOptions[key],
                                                        'data-funnelback-hidden': 'hidden'
                                                    }).appendTo(formID);                                            
                                                }
                                            }

                                            function globalSearchInit() {
                                                // Variables
                                                var navigationGlobalSearchToggleTrigger = $('[data-sitecore-js-trigger=navigation-global-search-toggle]');
                                                var navigationGlobalSearchToggleTarget = $('[data-sitecore-js-target=navigation-global-search-toggle]');
                                                var navigationGlobalSearchScope = '[data-sitecore-js-scope=navigation-global-search]';
                                                var siteSearchScopeIndex = 0;
                                                var hiddenInputs = $(navigationGlobalSearchScope).find('input[type=hidden]'); // count all input hidden instance inside scope

                                                var totalSiteSearchScopeIndex = hiddenInputs.length - 1; // as .eq() starts from 0; while .length starts from 1

                                                // Update Search Scope Hidden Fields
                                                function setData(counter, total) {
                                                    counter === total ? counter = 0 : counter++;
                                                    var action = hiddenInputs.eq(counter).val();
                                                    var label = hiddenInputs.eq(counter).attr('data-search-scope');
                                                    var nextLabel = counter + 1 > total ? hiddenInputs.eq(0).attr('data-search-scope') : hiddenInputs.eq(counter + 1).attr('data-search-scope'); // modify action attr on form

                                                    hiddenInputs.parent().attr('action', action); // modify labels

                                                    navigationGlobalSearchToggleTrigger.attr('title', "Toggle Search to ".concat(nextLabel));
                                                    navigationGlobalSearchToggleTarget.attr('title', "Search ".concat(label)).html("Search ".concat(label));
                                                    return counter;
                                                }

                                                // Event Triggers
                                                navigationGlobalSearchToggleTrigger.on('click', function (event) {
                                                    siteSearchScopeIndex = setData(siteSearchScopeIndex, totalSiteSearchScopeIndex);
                                                    updateHiddenInputsAndAction("navigation-global-search-form-id");
                                                });
                                            }

                                            window.onload = function () {
                                                globalSearchInit();
                                            };
