        function refocus() {
            $('#input').focus();

            $('#divider').html(fl);
        };

        function post_callback(data) {
            $('#input').val(data[0]);
            $('#input').select();
            if (data.length == 2 && data[1]) {
                $('#not_changed').show();
            };
            $('#click').val('✂'); $('#click').show();
            $('.descp2').show();
            $("#qrimage").attr('src','http://disk.yandex.net/qr/?clean=1&text=' + data[0]);
            filldivider('<a href="/">Добавить еще одну ссылку</a>');
            $('#form').submit( function () {
                $('#input').select();
                console.log($('#input').val());
                document.execCommand("copy");
                return false;
            });
            Ya.share({
                elementID: 'ya_share1',
                link: data[0]
            });
        };

        function error_callback() {
            var message = $('#message');
            message.text('Ссылка не соответствует ограничениям сервиса');
        };

        function put_url() {
            $('.descp').hide();
            filldivider(' ');
            $('#click').hide();
            $('#form').unbind();
            $.ajax({
                dataType: "json",
                url: '/--?json=true&url=' + encodeURIComponent($('#input').val()),
                success: post_callback,
                error: error_callback
            });
            return false;
        };

        $(document).ready(function() {
            refocus();
            $('#form').submit(put_url);
        });
    })();
