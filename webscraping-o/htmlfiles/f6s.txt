   function csAuthHelper(authContext) {
        this.authPopupUri = '/main/authorization/login?popup=true';
        if ('object' == typeof authContext) {
            this.authPopupUri += '&context=' + encodeURIComponent(JSON.stringify(authContext));
        }

        this.forgotPasswordUri = '/main/authorization/popup-forgot-password';
        return this;
    }

    csAuthHelper.prototype.removeURLHost = function(url) {
        if( 'undefined' == typeof( url ) ) return '';
    
        if( '' == $.trim( url ) ) return url;
        var substrings = [ cs.options.site_url, cs.options.site_url_ssl ];
        for( i in substrings ) {
            var substring = substrings[ i ];
            if( url.substring( 0, substring.length ) == substring ) {
                url = url.substring( substring.length );
            }
        }
        return url;
    }

            csAuthHelper.prototype.loginFB = function(return_url, skip_welcome) {
            if ('boolean' != typeof skip_welcome) {
                skip_welcome = false;
            }

            if( 'undefined' == typeof return_url || '' == $.trim( return_url ) ) {
                return_url = this.queryReturnUrl() || window.location.href;
            }

            return_url = this.removeURLHost( return_url );

            csFacebook.login({
                'return_url': return_url,
                'skip_welcome': skip_welcome
            });

            // Track the click of the FB button on Google Analytics
            if ('undefined' != typeof _gaq) {
                _gaq.push(['_trackEvent', 'Facebook Login', 'Initiate OAuth']);
            }
        }

    csAuthHelper.prototype.loginEmail = function(returnUrl) {
        if ('undefined' == typeof returnUrl || '' == $.trim(returnUrl)) {
            returnUrl = this.queryReturnUrl() || window.location.href;
        }

        csPopup.display(this.authPopupUri, {
            customClassName: 'tight rounded',
            postObject: {
                state:      'login',
                return_url: returnUrl
            },
        });
    }

    csAuthHelper.prototype.register = function(params) {
        var loginPopupUrl = this.authPopupUri + '&state=register';

        if (
            'object' == typeof params &&
            null !== params &&
            !Array.isArray(params) &&
            0 < Object.keys(params).length
        ) {
            loginPopupUrl += '&' + $.param(params);
        }

        csPopup.display(loginPopupUrl, {
            customClassName: 'tight rounded'
        });
    }

    csAuthHelper.prototype.forgotPassword = function() {
        csPopup.display(this.forgotPasswordUri, {
            customClassName: 'tight rounded'
        });
    }

    csAuthHelper.prototype.loginPopup = function() {
        var queryReturnUrl = this.queryReturnUrl();
        if ('string' === typeof queryReturnUrl) {
            queryReturnUrl = this.removeURLHost(queryReturnUrl);
        }

        csPopup.display(this.authPopupUri, {
            postObject: (queryReturnUrl ? {
                return_url: queryReturnUrl,
            } : undefined),
            customClassName: 'tight rounded'
        });
    }

    csAuthHelper.prototype.promptLogin = function(loginMethod, returnUrl) {
        if ('string' != typeof loginMethod) return;
        if ('string' != typeof returnUrl) returnUrl = null;

        switch (loginMethod) {
            case 'fb':
                return this.loginFB(returnUrl);
            case 'li':
                return this.loginLI(returnUrl);
            case 'weibo':
                return this.loginWeibo(returnUrl);
            case 'email':
                return this.loginEmail(returnUrl);
            default:
                return this.loginPopup();
        }
    }

    /**
     * Identify any post-login redirect instructions passed in the URL
     */
    csAuthHelper.prototype.queryReturnUrl = function() {
        var queryStringParams = ((window.___cs || {}).$utils || {}).queryStringParams;
        if ('function' !== typeof queryStringParams) {
            // Logic to extract the redirection URL has not been loaded yet
            return;
        }

        var queryParams = queryStringParams();
        if ('string' !== typeof queryParams.returnUrl || '' === queryParams.returnUrl.trim()) {
            // There is no redirection information in the URL
            return;
        }

        // Found the URL needed to redirect to
        return queryParams.returnUrl;
    }
