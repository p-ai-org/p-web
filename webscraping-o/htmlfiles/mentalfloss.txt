      function slide_hero( direction ){            // HERO SLIDESHOW
        var nextId = jQuery(".hero-next").attr("id").match(/\d+/);
        var currentId = jQuery(".hero-current").attr("id").match(/\d+/);
        var previousId = jQuery(".hero-previous").attr("id").match(/\d+/);

        jQuery(".hero-" + direction).hide();    // LOOPS SLIDESHOW
        jQuery(".hero-" + direction).insertBefore(jQuery(".hero-current"));
        jQuery(".hero-" + direction).show();

        jQuery("#hero-article-" + nextId).removeClass("hero-next");
        jQuery("#hero-article-" + currentId).removeClass("hero-current").addClass("hero-"+ direction);
        jQuery("#hero-article-" + previousId).removeClass("hero-previous");

        if(direction == "next") {
            jQuery("#hero-article-" + nextId).addClass("hero-previous");
            jQuery("#hero-article-" + previousId).addClass("hero-current");
        } else if(direction == "previous"){
            jQuery("#hero-article-" + nextId).addClass("hero-current");
            jQuery("#hero-article-" + previousId).addClass("hero-next");
        }

        jQuery(".hero-index").removeClass('active-index'); // UPDATE INDEX DOTS
        jQuery("#hero-index-" + jQuery(".hero-current").attr("id").match(/\d+/) ).addClass('active-index');
    }

            elem = equalizerContainer.find('.equalizer-watch'),
            maxHeight = 0;

        if (MentalFloss.getCurrentDevice() == "desktop" || MentalFloss.getCurrentDevice() == "tablet") {

            elem.each(function(){

                $(this).height('auto');

                if ($(this).height() > maxHeight) {
                    maxHeight = $(this).height();
                }
            });

            elem.height(maxHeight);

        } else {

            elem.height('auto');
        }
    }

      function initPlaylist() {
            var playlistId = "01dptcpgvzz5y8y384";
            var playlistUrl = "https://vms-videos.minutemediaservices.com/" + playlistId + "/" + playlistId + ".json";
            jQuery.ajax({
                url: playlistUrl,
                success: function(response) {
                    var playlistWrapper = jQuery('.playlist-wrapper');
                    var playlistTemplate = Handlebars.compile(jQuery('#out-player-playlist-template').html());
                    playlistWrapper.html(playlistTemplate(response));
                    numberVideos = response.data.length;
                    jQuery('#player-playlist ul.playlist').css('width', numberVideos * 310 +'px');
                    for(var i=0; i<response.data.length; i++) {
                        var vid = response.data[i].payload_id;
                        if(i===0) setVideo(vid);
                        jQuery(document).on('click', "a#playlist-item-link-" + vid, function(e) {
                            e.preventDefault();
                            vid = this.dataset.mediaid;
                            setVideo(vid);
                            setActive(vid);
                            return false;
                        });
                    }
                }
            });

        function setVideo(videoId) {
            let playerContainer = document.getElementById("player-container");
            let playerWrapper = document.getElementById("player-wrapper");
            let newWrapper = document.createElement("div");
            newWrapper.id = "player-wrapper";
            playerContainer.removeChild(playerWrapper);

            window.monti().initiate({
                "player_pos": newWrapper,
                "player_id": "01dnevbq6gva107mjc",
                "content_id": videoId
            });
        }

        function setActive(videoId) {
            jQuery('.playlist-item-link').removeClass('active');
            jQuery('#playlist-item-link-'+videoId).addClass('active');

            if(MentalFloss.getCurrentDevice() == 'desktop' || MentalFloss.getCurrentDevice() == 'tablet') {
                scrollOffToVideo = jQuery('li#item-' +videoId).prev().offset().left - 70;
                var leftCurrentPos = jQuery('.playlist-wrapper').scrollLeft();
                jQuery(".playlist-wrapper").animate({scrollLeft: leftCurrentPos + scrollOffToVideo}, 800);
            }


        }

                if (lock_studios_slide == false) {
                    lock_studios_slide = true;
                    videoItemWidth =  jQuery('li.playlist-item').first().outerWidth( true );

                    if (direction == "previous") {
                        var leftPos = jQuery('.playlist-wrapper').scrollLeft();
                        jQuery(".playlist-wrapper").animate({scrollLeft: leftPos - videoItemWidth}, 800);
                    }

                    if (direction == "next") {
                        var leftPos = jQuery('.playlist-wrapper').scrollLeft();
                        jQuery(".playlist-wrapper").animate({scrollLeft: leftPos + videoItemWidth}, 800);
                    }
                    setTimeout(function () {
                        lock_studios_slide = false;
                    },400);
                }
