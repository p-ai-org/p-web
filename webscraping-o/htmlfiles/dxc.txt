                    $(target).dropdown('toggle');
                }
            }

        $('#navbar-dropdown-backdrop').on('mouseenter focus', function() {
            $('.dropdown-menu.navbar-dropdown-menu').each(function() {

                    var target = '#' + $(this).attr('id').replace('-container', '');
                    // NOTE: .('hide') is Bootstrap 4.2 only
                    // $(target).dropdown('hide');
                     $(target).dropdown('toggle');
                }
            });
        });

        $(target).parent().on('show.bs.dropdown', function() {
            $('#navbar-dropdown-backdrop').addClass('show');
        });
        $(target).parent().on('hide.bs.dropdown', function() {
            $('#navbar-dropdown-backdrop').removeClass('show');
        });

        // @DEVNOTE: Support for legacy DXC templates
        if($.fn.dropdown.Constructor.VERSION.charAt(0) == '3') {
            $(target).parent().on('show.bs.dropdown', function() {
                $(target + '-container').addClass('show');

                $(target + '-container').removeClass('show');
                $('#navbar-dropdown-backdrop').removeClass('show');
            });
        }
