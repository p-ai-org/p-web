var namesMatch = function(name, vname){
    var name1 = name.replace(/[\W_]+/g, ' ').trim().toLowerCase();
    var name2 = vname.replace(/[\W_]+/g, ' ').trim().toLowerCase();
    if(!name1 || !name2) return false;

var findMatchingVideo = function(vp, vid){
    var pl = vp.getActualPlaylist();
    vid = vid.replace(/\/+$/,'').replace(/%20/g, ' ');
    if(vid.match(/^\d+$/)){
        var number = parseInt(vid);
        return pl[number-1];
    }else if(vid.match(/^\w\w-/)){
        var s = sources[vid.substr(0, 2)];
        for(var v in pl){
            if(pl[v].source == s && pl[v].video_id == vid.substr(3)){
                return pl[v];
            }
        }
    }else{
        var bm = 0;
        var winner = -1;
        for(var i=0; i<pl.length; i++){
            var candidate = namesMatch(vid, pl[i].title);
            if(candidate > bm){
                bm = candidate;
                winner = i;
            }
        }
        if(winner == -1) return null;

        if(window.console && console.log)
            console.log('Found video: ' + winner + ' for query ' + vid);
        return pl[winner];
    }
}

var parseAction = function(vp, x){
    var video = null;
    var start = 0;
    var stop = null; // at the end of video
    if(x.match(/\//)){
        var vid_pos = x.split(/\//, 2);
        x = vid_pos[0];
        var positions = vid_pos[1].split(/\-|\.\./, 2);
        start = parsePosition(positions[0]) || 0;
        if(positions[1] && positions[1].trim()){
            stop = parsePosition(positions[1]) || null;
        }
    }
    video = findMatchingVideo(vp, x);
    return {
        'video': video,
        'start': start,
        'stop': stop
    };
};

            Situation (Sartre)
           </h1>
           <p id="mwAQ">
            One of the first times in which
            <a href="https://wn.com/Jean-Paul_Sartre" id="mwAg" title="Jean-Paul Sartre">
             Jean-Paul Sartre
            </a>
            discussed the concept of
            <i id="mwAw">
             <b id="mwBA">
              situation
             </b>
            </i>
            was in his 1943
            <i id="mwBQ">
             <a href="https://wn.com/Being_and_Nothingness" id="mwBg" title="Being and Nothingness">
              Being and Nothingness
             </a>
            </i>
            , where he famously said that
           </p>
           <p id="mwCA">
            Earlier in 1939, in his short story
            <i id="mwCQ">
             <a href="https://wn.com/The_Wall_(short_story_collection)" id="mwCg" title="The Wall (short story collection)">
              The Childhood of a Leader
             </a>
            </i>
            , collected in his famous
            <i id="mwCw">
             <a href="https://wn.com/The_Wall_(book)" id="mwDA" title="The Wall (book)">
              The Wall
             </a>
            </i>
            , referring to a fake
            <a href="https://wn.com/Turd" id="mwDQ" title="Turd">
             turd
            </a>
            , he said that in
            <a href="https://wn.com/Prank" id="mwDg" title="Prank">
             pranks
            </a>
            "There is more destructive power in them than in all the works of Lenin."
Another famous use of the term was in 1945, in his editorial of the first issue of
            <i id="mwDw">
             <a href="https://wn.com/Les_Temps_modernes" id="mwEA" title="Les Temps modernes">
              Les Temps modernes
             </a>
            </i>
            (Modern Times); arguing the principle of the responsibility of the intellectual towards his own times and the principle of an engaged literature, he summarized: "the writer is in a situation with his epoch."
           </p>
           <p id="mwEQ">
            An, influential use of the concept was in the context of theatre, in his 1947 essay
            <i id="mwEg">
             For a Theatre of Situations
            </i>
            . A passage that has been frequently quoted is the following, in which he defines the
            <i id="mwEw">
             Theater of Situations
            </i>
            :
           </p>
           <p id="mwFQ">
            He then published his series
            <i id="mwFg">
             Situations
            </i>
            , with ten volumes on
            <i id="mwFw">
             Literary Critiques
            </i>
            and
            <i id="mwGA">
             <a href="https://wn.com/What_Is_Literature?" id="mwGQ" title="What Is Literature?">
              What Is Literature?
             </a>
            </i>
            (1947), the third volume (1949),
            <i id="mwGg">
             Portraits
            </i>
            (1964),
            <i id="mwGw">
             <a href="https://wn.com/Colonialism_and_Neocolonialism" id="mwHA" title="Colonialism and Neocolonialism">
              Colonialism and Neocolonialism
             </a>
            </i>
            (1964), Problems of Marxism, Part 1 (1966),  Problems of Marxism, Part 2 (1967),
            <i id="mwHQ">
             The Family Idiot
            </i>
            (1971-2),
            <i id="mwHg">
             Autour de 1968
            </i>
            and
            <i id="mwHw">
             Melanges
            </i>
            (1972), and
            <i id="mwIA">
             Life/Situations: Essays Written and Spoken
            </i>
            (1976).
