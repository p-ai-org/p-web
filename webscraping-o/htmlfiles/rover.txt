   window.addEventListener("load", function(event) {
    function browserSupportsPerformanceMetrics() {
      var timing = window.performance && window.performance.timing;
      var paintEntries = window.performance && window.performance.getEntriesByType && window.performance.getEntriesByType('paint');
      var longTaskTiming = window.PerformanceObserver && window.PerformanceLongTaskTiming;
      return timing || paintEntries || longTaskTiming;
    }

    // abstracts timing data so that it guarantees returning an object
    function getTimingData() {
      var metrics = { event: 'pageview-performance' };
      var timing = window.performance && window.performance.timing;
      if (!timing) {
        return metrics;
      }
      var timingKeys = [
          "unloadEventStart", "unloadEventEnd", "redirectStart", "redirectEnd",
          "fetchStart", "domainLookupStart", "domainLookupEnd", "connectStart",
          "connectEnd", "secureConnectionStart", "requestStart", "responseStart",
          "responseEnd", "domLoading", "domInteractive", "domContentLoadedEventStart",
          "domContentLoadedEventEnd", "domComplete", "loadEventStart", "loadEventEnd"
        ];
      var start = timing.navigationStart;
      timingKeys.forEach(function(k) {
        var time = window.performance.timing[k];
        if (time > 0) {
          metrics[k] = time - start;
        }
      });
      return metrics;
    }

    // camelizes the event entry names in the PeformancePaintTiming objects. Gaurantees returning a list.
    function getPaintData() {
      var entries = window.performance && window.performance.getEntriesByType && window.performance.getEntriesByType('paint') || [];
      var paintDataKeyMap = {
        'first-paint': 'firstPaint',
        'first-contentful-paint': 'firstContentfulPaint'
      };
      return entries.map(function(entry) {
        return {
          name: paintDataKeyMap[entry.name],
          timing: parseInt(entry.startTime, 10),
        }
      })
    }

    setTimeout(function() {
      if ( !browserSupportsPerformanceMetrics() ) {
        // Just push an empty object if the browser doesn't have any performance metrics.
        // This means we can use the fact that this event triggers at all for a proxy event
        // in our bounce rate metrics.
        dataLayer.push({ event: 'pageview-performance' });
        return;
      }

      var performanceMetrics = getTimingData();
      getPaintData().forEach(function(paintEntry) {
        performanceMetrics[paintEntry.name] = paintEntry.timing;
      });

      ttiPolyfill.getFirstConsistentlyInteractive().then((tti) => {
        // N.B.: tti is null on browsers that don't support the metrics for it.
        if (tti) {
          performanceMetrics['timeToInteractive'] = Math.round(tti);
        }
        dataLayer.push(performanceMetrics);
      });
    }, 0);
  });
