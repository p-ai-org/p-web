     <script type="text/javascript">
      var links_array = new Array();
    var url_GetLink = '/filesystem/GetLink';

    function openTab(evt, cityName) {
        var i, tabcontent, tablinks;

        tabcontent = document.getElementsByClassName("tabcontent");
        for (i = 0; i < tabcontent.length; i++) {
            tabcontent[i].style.display = "none";
        }

        tablinks = document.getElementsByClassName("tablinks");
        for (i = 0; i < tablinks.length; i++) {
            tablinks[i].className = tablinks[i].className.replace(" active", "");
        }

        document.getElementById(cityName).style.display = "block";
        evt.currentTarget.className += " active";
    }

    function changeLinkType()
    {
        makeLinks(links_array);
    }

    function makeLinks(arr)
    {
        var domain = dl_path, name = '';
        var domain_rgto = 'http://rg.to';
        domain_rgto = domain_rgto.replace('http:', document.location.protocol);
        var names = arr[0], links = arr[1], sizes = arr[2], is_shrinking = arr[3], ids = arr[4];
        var text = '', html = '', bbcode = '';
        if (links != null) {
            links.forEach(function (e, i) {
                name = '';
                if ($('#link_type').val() != 3) {
                    if ($('#wo_name').is(':checked') && sizes[i] != '') {
                        name = '';
                    } else {
                        name = '/' + names[i] + '.html';
                    }
                }
                if (sizes[i] == '') {
                    typeLink = 'folder';
                    name = encodeURI(name);
                } else {
                    typeLink = 'file';
                }
                if ($('#link_type').val() == 2) {
                    e = domain + '/' + typeLink + '/' + ids[i];
                }
                if ($('#link_type').val() == 1) {
                    e = domain_rgto + '/' + typeLink + '/' + ids[i];
                }
                if ($('#link_type').val() == 3) {
                    $('#woname').hide();
                } else {
                    $('#woname').show();
                }
                size = '';
                if ($('#file_size').is(':checked') && sizes[i] != '') {
                    size = ' (' + sizes[i] + ')';
                }
                var nl = "\r\n";
                text = text + e + name + nl;
                html = html + '&lt;a href="' + e + name + '"&gt;' + names[i] + '&lt;/a&gt;' + nl;
                bbcode = bbcode + '[url=' + e + name + ']' + names[i] + '[/url]' + size + nl;
            });
        }
        $("#text").find('textarea').html(text);
        $("#html").find('textarea').html(html);
        $("#bbcode").find('textarea').html(bbcode);
        if (is_shrinking) {
            if ($("#link_type option[value='3']").length == 0) {
                $('#link_type').append(new Option('short (ad.to)', 3, true, true));
                changeLinkType();
            }
        }
    }

    function copy()
    {
        try {
            var id = ($('#mytabs').find('.tablinks.active').attr('name')).toLowerCase();
            $('#'+id).find('textarea').select();
            document.execCommand('copy');
        } catch (err) {
            alert('Unable to copy! Unsupported Browser!');
        }
    }

    function download()
    {
        href = "data:text/plain;charset=utf-8,";
        href = href+encodeURIComponent('[Plain]'+"\r\n"+$("#text").text().replace(/<br>/g,"\r\n")+"\r\n");
        href = href+encodeURIComponent('[Html]'+"\r\n"+$("#html").text().replace(/></g,">\r\n<")+"\r\n\n");
        href = href+encodeURIComponent('[BBcode]'+"\r\n"+$("#bbcode").text().replace(/<br>/g,"\r\n")+"\r\n");
        $("#dwnl_links").attr('href', href);
        $('#dwnl_links').get(0).click();
    }

    $(document).ready(function() {
        $('#file_size').change(function () {
            changeLinkType();
        });

        $('#wo_name').change(function () {
            changeLinkType();
        });

        document.getElementById("defaultOpen").click();
    });
