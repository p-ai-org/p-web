function createCookie(name, value, days, path, domain) {
if (name.length == 0) {
    return
}
var expires;
if (days) {
    var date = new Date();
    date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
    expires = "; expires=" + date.toGMTString();
} else {
    expires = "";
}
  document.cookie = name + "=" + value + expires + "; path=" + path + "; domain=" + domain;
}

function getCookie(cname) {
  var name = cname + "=";
  var ca = document.cookie.split(';');
  for(var i=0; i<ca.length; i++) {
      var c = ca[i];
      while (c.charAt(0)==' ') c = c.substring(1);

        if (name !== "" && name !== null && name != undefined) {
            name = name.replace(/[\[]/, "\\[").replace(/[\]]/, "\\]");

                results = regex.exec(location.search);
            return results === null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));
        } else {
            var arr = location.href.split("/");
            return arr[arr.length - 1];
        }

    function getLiveIndicatorStatus() {
        jQuery.getJSON(iconLiveFeedURL, function(data) {
            var seriesID = data['series_id'];

            if(seriesID == "1" || seriesID == "2" || seriesID == "3"){
                var htmlText = jQuery('.race-center-footer-tab').html();
                var className='race-center-live-bg';
                lapsToGo = data["laps_to_go"];
                runType = data["run_type"];
                flagState = data['flag_state'];

                if(seriesID=="3"){
                    jQuery(".popup-item.drive").hide();
                }
                if(seriesID =="2" || seriesID=="3"){
                if(runType == "3" && lapsToGo <= 0){
                    raceCenterURL = seriesOneRC;
                }
                else{
                    if(seriesID=="2"){
                    raceCenterURL = seriesTwoRC;
                    }
                    else{
                    raceCenterURL = seriesThreeRC;
                    }
                }
                }

                if(data != null && data['flag_state'] !=''){
                    className = className+'-'+data['flag_state'];
                    if(isMobile && isHomepage && !disableTicker){
                        //Call Ticker Function when Enabled
                        buildNotificationBar(data, raceCenterURL);
                    }
                }

                jQuery(".race-center-footer-tab").attr("href",raceCenterURL);
                jQuery(".popup-item.leaderboard .text-container a").attr("href",raceCenterURL);
                if(data['flag_state'] !='' && data['flag_state'] !=9) {
                    jQuery('.footer-tab-text.race-center-flag').attr('class', 'footer-tab-text race-center-flag '+className);
                    if(jQuery(".live-notification-bar")[0] && isHomepage && !disableLiveFeed && !disableTicker){ //Ticker exists update flag state
                        jQuery('.live-race-indicator').attr('class', 'live-left-button live-race-indicator race-center-live-'+flagState);

                    if (htmlText.indexOf('Race Center') > -1) {

                        jQuery('.race-center-footer-tab').html(htmlText);
                    }
                }

    function buildNotificationBar(liveData, raceCenterURL){
        if(liveData['flag_state'] !='' && liveData['flag_state'] !=9){
            var raceID = liveData['race_id'];
            var ctaButtonText = 'Race Center';
            var raceType;
            var ticker_seriesID = liveData['series_id'];
            var eventRaceName = "";
            var htmlLeftBtn = '', html = '', htmlRightBtn = '';

            //Use Editorial manual override configurations
            if(rightButtonURL){
                var raceCenterURL = rightButtonURL;
            }
            if(rightButtonText){
                ctaButtonText = rightButtonText;
            }
            if(centerText){
                eventRaceName = centerText;
            }

            if(runType == "1"){
                raceType = "Practice";
            } else if (runType == "2"){
                raceType = "Qualifying";
            } else if (runType == "3"){
                raceType = "Race";
            }

            var lapsTotal = liveData['laps_in_race'];
            var lapsComplete = liveData['lap_number'];
            var stagesSeries1 = null;
            var stagesSeries2 = null;
            var stagesSeries3 = null;

            var totalStages = "";

            switch(ticker_seriesID){
                case '1':
                    totalStages = stagesSeries1;
                    break;
                case '2':
                    totalStages = stagesSeries2;
                    break;
                case '3':
                    totalStages = stagesSeries3;
                    break;
                default:
                    totalStages = stagesSeries1;
                    break;
            }

            if(!jQuery(".live-notification-bar")[0] ){
                //left Button
                if(disableLiveFeed){
                    htmlLeftBtn = '<div class="live-left-button live-race-indicator">'+
                    '</div>';
                } else {
                    htmlLeftBtn = '<div class="live-left-button live-race-indicator race-center-live-'+flagState+'">'+
                        '<i class="fa fa-lg fas fa-flag-checkered"></i>'+
                        '<span class="text">LIVE</span>'+
                    '</div>';
                }
                //event details
                if(disableLiveFeed){
                    htmlEventDetails = '<div class="live-event-details" tabindex="1">' +
                        '<div class="live-event-name">'+eventRaceName+'</div>' +
                    '</div>';
                } else {
                    htmlEventDetails = '<div class="live-event-details" tabindex="1">' +
                        '<div class="live-event-name">'+eventRaceName+'</div>' +
                        '<div class="live-event-laps" style="display:none;">Laps: <span>'+lapsComplete+'/'+lapsTotal+'</span></div>' +
                        '<div class="live-event-type"><span style="display:none;">'+raceType+'</span></div>' +
                        '<div class="live-event-stage" style="display:none;">Stage: <span>'+liveData['stage']['stage_num']+' of '+(eval(totalStages).filter(Boolean).length + 1)+'</span></div>' +
                    '</div>';
                }

                //right button
                htmlRightBtn = '<a class="live-right-button" onclick="rightBtn()" href="'+raceCenterURL+'">'+
                        '<span>'+ctaButtonText+'</span>'+
                    '</a>';

                htmlData = '<div class="live-notification-bar width-1200px">' + htmlLeftBtn + htmlEventDetails + htmlRightBtn + '</div>';

                function addTicker(htmlData){
                    jQuery('.header').append(htmlData);
                    var tickerHeight = jQuery(".live-notification-bar ").height();
                    jQuery(".live-notification-bar ").parent().next().css("padding-top", (tickerHeight-6)+"px");//TO DO: Remove the additional 6 paddings once Mweb redesign launches
                    if(jQuery(".mobile-featured-container .mobile-featured-buttons")[0] ){
                        var tabsAdjustment = tickerHeight + 40;
                        var tabsHeight = jQuery(".mobile-featured-container .mobile-featured-buttons").height();
                        jQuery(".mobile-featured-inner-container-items").css("margin-top", tabsHeight+"px");
                        jQuery(".mobile-featured-container .mobile-featured-buttons").css("top", tabsAdjustment+"px");
                    }
                }

                if(!disableTicker){
                    if(!eventRaceName){
                        jQuery.ajax({
                            type: "GET",
                            url: '/wp-admin/admin-ajax.php',
                            data: {
                                action: 'global_events_lookup',
                                meta_query: {"eventType": "schedule", "eventSeries":ticker_seriesID, "eventRaceID":raceID}
                            },
                            success: function (json) {
                                var event_data = JSON.parse(json);
                                console.log("event data", event_data);
                                eventRaceName = event_data['eventName'];
                                addTicker(htmlData);
                                jQuery('.live-event-name').text(eventRaceName);
                                if(runType == "3"){
                                    jQuery('.live-event-type span').hide();
                                    jQuery('.live-event-stage').show();
                                    jQuery('.live-event-laps').show();
                                    jQuery('.live-event-type').addClass("separator");

                                } else {
                                    jQuery('.live-event-type span').show();
                                    jQuery('.live-event-stage').hide();
                                    jQuery('.live-event-laps').hide();
                                }
                            },
                            error: function (xhr, status, error) {
                                var err = eval("(" + xhr.responseText + ")");
                                console.log("err", err);
                                addTicker(htmlData);
                            }
                        });
                    } else {
                        addTicker(htmlData);
                    }

                }

            }

            if(jQuery(".live-notification-bar")[0] && !disableLiveFeed ){ //Already exist only update details that can change
                jQuery('.live-race-indicator').attr('class', 'live-left-button live-race-indicator race-center-live-'+flagState);
                if(runType == "3"){
                    jQuery('.live-event-type span').hide();
                    jQuery('.live-event-type').addClass("separator");
                    jQuery('.live-event-stage').show();
                    jQuery('.live-event-stage').html('Stage: <span>'+liveData['stage']['stage_num']+' of '+(eval(totalStages).filter(Boolean).length + 1)+'</span>');
                    jQuery('.live-event-laps').show();
                    jQuery('.live-event-laps').html('Laps: <span>'+lapsComplete+'/'+lapsTotal+'</span>');

                } else {
                    jQuery('.live-event-type span').show();
                    jQuery('.live-event-type').removeClass("separator");
                    jQuery('.live-event-stage').hide();
                    jQuery('.live-event-laps').hide();
                }
            }
        }else{
            jQuery('.live-notification-bar').remove();
            jQuery(".live-notification-bar ").parent().next().prop("padding-top");
            if(jQuery(".mobile-featured-container .mobile-featured-buttons")[0] ){
                jQuery(".mobile-featured-container .mobile-featured-buttons").css("top", "40px");
                jQuery(".mobile-featured-inner-container-items").css("margin-top", "");
            }
        }
}
