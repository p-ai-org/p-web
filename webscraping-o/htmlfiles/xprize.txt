      //$(document).ready(function () {
		//We check if the localStorage has the prizes for slideout. If yes then we check expiry date for that data. If not expired, we show
		//data from localStorage which is saved as JSON. If expired, we make a call to the backend and refresh data and update expiry to 12 hours.
	var pagesKey = "visioneeringSubPages";
	$(document).ready(function () {
		
			if (helper.LocalDataExpired(pagesKey)) {
				getVisioneeringSubPages();
			} else {
				renderVisioneeringSubPagesForMenu(localStorage.getItem(pagesKey));
			}
	});

	function getVisioneeringSubPages() {
		helper.HTTP_GET('/Visioneering/GetVisioneeringSubPages', function (data) {
			if (data !== false) {
				renderVisioneeringSubPagesForMenu(data);
				helper.SetLocalData(pagesKey, data);
			}
		});
	}

	function renderVisioneeringSubPagesForMenu(json) {
		var pages = JSON.parse(json);
		for (var i = 0; i < pages.length; i++) {
			document.getElementById('visioneering-menuitems').innerHTML +=
				'<li class="menu__item">'
				+ '<a class="menu__link submenuitem hover-pink" href="/visioneering/page/' + pages[i].slug + '">' + pages[i].title + '</a>'
				+ '</li>';
		}
	}

	function openVisioneeringSubPages() {
			$('#visioneering-menuitems').slideToggle();
		}

   $(document).ready(function () {

    $(document).ready(function () {

        var animating = false;
        var cardsCounter = 0;
        var numOfCards = 6;
        var decisionVal = 80;
        var pullDeltaX = 0;
        var deg = 0;
        var $card;

        function pullChange() {
            animating = true;
            deg = pullDeltaX / 10;
            $card.css("transform", "translateX(" + pullDeltaX + "px) rotate(" + deg + "deg)");
        };

        function release() {

            if (pullDeltaX >= decisionVal) {
                $card.addClass("to-right");
            } else if (pullDeltaX <= -decisionVal) {
                $card.addClass("to-left");
            }

            if (Math.abs(pullDeltaX) >= decisionVal) {
                $card.addClass("inactive");

                setTimeout(function () {
                    $card.addClass("below").removeClass("inactive to-left to-right");
                    cardsCounter++;
                    if (cardsCounter === numOfCards) {
                        cardsCounter = 0;
                        $(".demo__card").removeClass("below");
                    }
                }, 300);

                $('.demo__card-cont').children().each(function (index) {
                    let currentIndex = +$(this).attr('data-index');
                    let newIndex = currentIndex + 1;
                    if (currentIndex === numOfCards - 1) {
                        newIndex = 0;
                    }
                    $(this).attr('data-index', newIndex);
                })
            }

            if (Math.abs(pullDeltaX) < decisionVal) {
                $card.addClass("reset");
            }

            setTimeout(function () {
                $card.attr("style", "").removeClass("reset")
                    .find(".demo__card__choice").attr("style", "");

                pullDeltaX = 0;
                animating = false;
            }, 300);

        };

        if ($(window).width() < 768) {

            $(document).on("mousedown touchstart", ".demo__card:not(.inactive)", function (e) {
                if (animating) return;

                $card = $(this);
                var startX = e.pageX || e.originalEvent.touches[0].pageX;

                $(document).on("mousemove touchmove", function (e) {
                    var x = e.pageX;
                    pullDeltaX = (x - startX);
                    if (!pullDeltaX) return;
                    pullChange();
                });

                $(document).on("mouseup touchend", function () {
                    $(document).off("mousemove touchmove mouseup touchend");
                    if (!pullDeltaX) return; // prevents from rapid click events
                    release();
                });
            });

            let isDragging = false;

            $(".general-card").on({
                mousedown: function (e) {
                    $(this).data({ top: e.pageX, left: e.pageY });
                },
                mouseup: function (e) {
                    var top = e.pageX,
                        left = e.pageY,
                        ptop = $(this).data('top'),
                        pleft = $(this).data('left');

                    if (ptop - top !== 0 || pleft - left !== 0) {
                        isDragging = true;
                    }
                },
                click: function (e) {
                    if (isDragging) {
                        e.preventDefault();
                        isDragging = false;
                    }
                }
            });
        }
    });

    $(document).ready(function () {
        $(document).scroll(function (evt) {
            let frontCard = $('.demo__card:last-child a');
            let v2 = Math.abs($(frontCard).position().top - $(window).height() / 2);
            let v1 = $(this).scrollTop();
            if (v1 > v2) {
                $(frontCard).addClass('start')
            } else {
                $(frontCard).removeClass('start');
            }
        });
    });
