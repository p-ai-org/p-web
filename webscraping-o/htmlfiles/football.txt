     <script>
      function valideEmail(){
        var mail = document.getElementById("mail");
        var isvalid = validateRegexEmail(mail.value);
        var message = document.getElementById("message");

        if(isvalid){ // valid

            message.innerHTML = "Thank You"
            message.classList.remove('alert-danger', 'd-none');
            message.classList.add('alert-success');
            mail.value = "";
            // hide the msg after 2sec
            setTimeout(function(){$('#message').hide(300);}, 2000);

            // send mail to server
            var additionalData = {};
            additionalData.application = 'ftbc-email';
            additionalData.module = 'ftbc-email';
            additionalData.guid = mail.value;
            WzAdMgr.track('acquisition', 'email_widget', additionalData);
        }
        else{ //invalid
            message.innerHTML = getErrorMsg(mail.value);
            message.classList.remove('alert-success', 'd-none');
            $('#message').show(300);
            message.classList.add('alert-danger');
        }
    }

    function validateRegexEmail(email) {
        var re = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
        return re.test(String(email).toLowerCase());
    }

    function getErrorMsg(mail){
        if(mail.indexOf('@') === -1){
            return "Please Put a Valid Email";
        }
        else if(mail.indexOf('@') === 0){
            return "Complete The First Part Of Your Mail please";
        }
    }
