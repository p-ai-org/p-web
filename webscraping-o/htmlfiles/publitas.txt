     function setCookie(cname, cvalue, exdays = 30) {
        var d = new Date();
        d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
        var expires = "expires="+d.toUTCString();
        var cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
        document.cookie = cookie;
        return cookie;
      }

      function getCookie(cname) {
        var name = cname + "=";
        var ca = document.cookie.split(';');
        for(var i = 0; i < ca.length; i++) {
          var c = ca[i];
          while (c.charAt(0) == ' ') {
            c = c.substring(1);
          }
          if (c.indexOf(name) == 0) {
            return c.substring(name.length, c.length);
          }
        }
        return undefined;
      }

      function goToLocalizedPage(location) {
        if (location && document.location.href === location) return;
        document.location = location
      }

      function getSelectedLang(e) {
        var initial = e.target.href.replace(document.location.origin, '').split('/')[1]
        return initial !== '' ? initial : 'en'
      }

      var LANGUAGE_COOKIE = 'publitas.language'
      var currentLang = getCookie(LANGUAGE_COOKIE)

      if (currentLang) {
        var alt = document.querySelector(`link[hreflang=${currentLang}]`)
        alt && goToLocalizedPage(alt.getAttribute('href'));
      } else {
        var browserLang = navigator.language.split('-')[0]
