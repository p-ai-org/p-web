                                        return;
                                    }

                                        _searchQuery.deleteBackButtonClickEvent(e, element);
                                    } else {
                                        //appends space after the current query
                                        _searchQuery.appendSpaceAfterQuery(element);
                                    }

                                    _searchQuery.getCursorPosition(element);

                                }

    var _searchQuery = {
        //Keys 0 - 9
        isNumberKey: function (target) {
            return (target.which >= 48 && target.which <= 57);
        },
        //Keys a - z
        isLetterKey: function (target) {
            return (target.which >= 65 && target.which <= 90);
        },
        //Keys numpad0 - numpad9
        isNumPadKey: function (target) {
            return (target.which >= 96 && target.which <= 105);
        },
        // Keys *,+,-,/,decimal point
        isMathKey: function (target) {
            return (target.which >= 106 && target.which <= 111);
        },
        //Keys :,=,,,dash,.,/,[,],',\
        isOperatorKey: function (target) {
            return (target.which >= 186 && target.which <= 192) || (target.which >= 219 && target.which <= 222);
        },
        //Keys Backspace
        isBackspaceKey: function (target) {
            return (target.which === 8);
        },
        //Keys delete
        isDeleteKey: function (target) {
            return (target.which === 46);
        },
        //SA click events
        isSAClick: function (elementID, e) {
            return ((e.target.tagName === "INPUT" || e.target.tagName === "TEXTAREA") && e.target.id !== elementID);
        },
        //Click event for delete & back button
        deleteBackButtonClickEvent: function (e, element) {
            e.preventDefault();
            element.value = element.value.substring(0, element.value.length - 1);
            if (typeof element.selectionStart === "number") {
                var textLen = element.value.length;
                element.setSelectionRange(textLen, textLen);
            }
            element.focus();
        },

        //Appends space after query in search box
        appendSpaceAfterQuery: function (element) {
            element.value = element.value + " ";
            /*Append space to the query once focus is out of text box*/
            element.focus();
        },

        //Gets cursor position inside search box
        getCursorPosition: function (element) {
            /*get cursor position in input box if browser supports selectionStart */
            if (typeof element.selectionStart === "number") {
                if (element.selectionStart === 0 && element.selectionEnd === element.value.length) {
                    element.value = "";
                    element.focus();
                } else {
                    if (element.selectionStart === 0 || element.selectionStart >= element.value.length) {
                        var len = element.value.length;
                        element.setSelectionRange(len, len);
                    }
                }
            } else {
                /*get the cursor position in the input box if the browser does not have a support for selectionStart*/
                var cursorPos = 0;
                if (document.selection) {
                    var oSel = document.selection.createRange();
                    oSel.moveStart("character", -element.value.length);
                    cursorPos = oSel.text.length;
                }
                if (cursorPos === 0 || cursorPos >= element.value.length) {
                    element.value = element.value;
                    /*setSelectionRange will not work for IE version < 9*/
                }
            }
        },

        //Checks whether pressed key is eligible for the functionality
        isEligibleKey: function (e) {
            return ((e.which !== 0 && !e.ctrlKey && !e.metaKey && !e.altKey) &&
                    (_searchQuery.isNumberKey(e) || _searchQuery.isLetterKey(e) || _searchQuery.isNumPadKey(e) ||
                            _searchQuery.isMathKey(e) || _searchQuery.isOperatorKey(e) || (_searchQuery.isBackspaceKey(e) ||
                                    _searchQuery.isDeleteKey(e))));
        }
