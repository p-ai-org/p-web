            e.preventDefault();

                let btn_active = $(btn_click).parent('li').parent('ul');

                    $(this).removeClass('open');
                    btn_click.removeClass('active');
                } else {
                    $(this).addClass('open');
                    btn_click.addClass('active');
                }
            });

            $cur_btn = btn_click;
        }

            });

            function hide_text_for_mobile() {
                $('.section-cols article').each(function (iter, e1) {
                    var item_block_flag = true;
                    $(this).find('li, p').each(function () {
                        if (item_block_flag) {
                            item_block_flag = false;
                        } else {
                            if ($(this).attr("aria-hidden") == undefined) {
                                $(this).addClass('hide_block_article')
                            }
                        }
                    });

                    if ($(this).find('.hide_block_article').length > 0) {
                        $(e1).append("<div class='read_more_text' >Read more <span>&#8250;</span></div>");
                    }
                });
            }

            function hide_text_for_pc() {
                $('.section-cols article').find('.read_more_text').remove();
                $('.section-cols article').find('li, p').removeClass('hide_block_article');
            }

            if ($(window).width() < 620) { //мобильные
                hide_text_for_mobile();
                mob_vers = true;
                pc_vers = false;
            } else {
                hide_text_for_pc();
                mob_vers = false;
                pc_vers = true;
            }

            $(window).resize(function () {
                if ($(window).width() < 620) { //мобильные
                    if (!mob_vers) {
                        hide_text_for_mobile();
                        mob_vers = true;
                        pc_vers = false;
                    }
                } else {
                    if (!pc_vers) {
                        hide_text_for_pc();
                        mob_vers = false;
                        pc_vers = true;
                    }
                }
            });

        });
