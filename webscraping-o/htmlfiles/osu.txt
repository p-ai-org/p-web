   $(document).ready( function() {
        var $dialog_modal = $('#ui-id-1-contents'),

            closeDialog = function() {
                $dialog_modal.dialog('close');
                setTimeout(refocus($trigger), 500);

            restoreContent = function() {
                var $item = $('.cloned-item .interact_item').detach(),
                    $target = $('.interact_item-placeholder');
                if ($item) {
                    $target.after($item);
                    $target.remove();
                    $item = null;
                }
                initModal();
            },
            grabContent = function(item) {
                var $item = item || '',
                    $copy;
                if ($item) {
                    if (!item.hasClass('interact_item')) {
                        $item = item.closest('.interact_item');
                    }
                    $item.before('<div class="interact_item-placeholder"></div>');
                    $('.event_intercept').remove();
                    $copy = $item.detach();
                }
                if ($copy) {
                    $copy.appendTo('.cloned-item');
                    $copy = null;
                }
            },
            refocus = function(trigger) {
                var $focus = trigger || '',
                    $focused = false,
                    i = 0;
                do {
                    if ($focus && !$($dialog_modal).dialog('isOpen')) {
                        $($focus).attr('tabindex', '-1').focus();
                        $focused = true;
                    }
                    i++;
                    console.log('refocusing of ' + $focus + ' attempt ' + i + '/100');
                }
                while (i < 100 && !$focused);
            },
            initModal = function() {
                //invoke the dialog on page load.
                $dialog_modal.dialog({
                    autoOpen: false,
                    draggable: false,
                    modal: true,
                    beforeClose: function() {
                        restoreContent();
                    },
                    close: function () {
                        closeDialog();
                    },
                    width: $vw,
                    hide: {
                        duration: 500
                    },
                    position: {
                        me: 'center top',
                        at: 'center top',
                        collision: 'none'
                    }
                });
                if ($('.yt-video-container').length) {
                    $('.yt-video-container').attr('autofocus', true);
                    $('.yt-video-container iframe').after('<div class="event_intercept"></div>');
                } else {
                    $('[src^="https://www.youtube.com/embed/"]').wrap('<div class="yt-video-container"></div>').after('<div class="event_intercept"></div>');
                }
