    //************************************************
    //            Initialize Isotope Code
    //************************************************
    function initializeIsotope(params) {
        var qsRegex; //regex search string used globally by Isotope
        var ddlFilter; //dropdown list filter used globally by Isotope

        //ISOTOPE GRID SETTINGS & FUNCTIONALITY
        var $grid = $("." + params["gridContainerClass"]).isotope({
            itemSelector: "." + params["itemContainerClass"],
            layoutMode: "fitRows",
            filter: function () {
                var $this = $(this);
                var searchResult = qsRegex ? $this.text().match(qsRegex) : true;
                var buttonResult = ddlFilter ? $this.is(ddlFilter) : true;
                return searchResult && buttonResult;
            },
            percentPosition: true,
            masonry: {
              columnWidth: '.program-result-item'
            },
            getSortData: {
                name: '.' + params["searchTitleClass"]
            }
        });

        $grid.isotope({ sortBy: params["searchTitleClass"], sortAscending: true }); //sorts data on page load

        //SEARCH BOX FUNCTIONALITY
        var $quicksearch = $("#" + params["searchBoxId"]).keyup(debounce(function () {
            // Use value of search field to filter
            qsRegex = new RegExp($quicksearch.val(), "gi");
            $grid.isotope();
            updateLazyLoad();
            checkResults();
        }, 200));

        //DROPDOWN BOX FUNCTIONALITY
        if (params["dropdownId"] != null) {
            if (params["dropdownId"] == "ac-filter-btn") {
                acBtn();
            } else if (params["dropdownId"].indexOf("category-filter") >= 0) {
                plDropdown();
            }
        }
        

        //DEBOUNCE FUNCTION: Debounce so filtering doesn't happen every millisecond
        function debounce(fn, threshold) {
            var timeout;
            return function debounced() {
                if (timeout) {
                    clearTimeout(timeout);
                }

                function delayed() {
                    fn();
                    timeout = null;
                }

                timeout = setTimeout(delayed, threshold || 100);
            };
        }

        //CHECK RESULTS FUNCITON: If nothing found during search display "no results" message
        function checkResults() {
            var visibleItemsCount = $grid.data("isotope").filteredItems.length;

            //Update Aria label
            var previousCount = $(".counter").text().match(/\d+/);
            if (previousCount == "") { //default state
                $(".ac-listing-items").attr("aria-relevant", "removals");
            }
            else if (visibleItemsCount < previousCount) {
                $(".ac-listing-items").attr("aria-relevant", "removals");
            }
            else if (visibleItemsCount > previousCount) {
                $(".ac-listing-items").attr("aria-relevant", "additions");
            }

            //Update count label
            if (visibleItemsCount > 1 || visibleItemsCount == 0) {
                $(".counter").text(visibleItemsCount + " items");
                $(".counter").show();
            } else {
                $(".counter").text(visibleItemsCount + " item");
                $(".counter").show();
            }

            if (visibleItemsCount > 0) {
                $("." + params["noResultsClass"]).hide();
            } else {

        function plDropdown() {
            $("#" + params["dropdownId"]).change(function () {
                ddlFilter = $(this).find(':selected').attr('data-filter');
                var filterAttr = $(this).find(':selected').text();
                // set filter in hash
                location.hash = 'filter=' + encodeURIComponent(filterAttr);
                $grid.isotope();
                checkResults();
            });
        }

        function getHashFilter() {
            // get filter=filterName
            var matches = location.hash.match(/filter=([^&]+)/i);
            var hashFilter = matches && matches[1];
            return hashFilter && decodeURIComponent(hashFilter);
        }

        //Check for #filters on load
        $(document).ready(function () {
            var filterVar = getHashFilter();
            if (filterVar != 'All' && filterVar !=null){
                $("select option").filter(function () {
                    return $(this).text() == filterVar;
                }).attr('selected', true);
                //initiate the filter
                ddlFilter = $(this).find(':selected').attr('data-filter');
                var filterAttr = $(this).find(':selected').text();
                // set filter in hash
                location.hash = 'filter=' + encodeURIComponent(filterAttr);
                $grid.isotope();
                checkResults();
            }
        });

        //AROUND CAMPUS FILTER BUTTON FUNCTION:
        function acBtn() {
            $("#" + params["dropdownId"]).click(function () {
                var filterName = this.getAttribute("data-filter");
                if (filterName == "AZ") {
                    $grid.isotope({ sortBy: params["searchTitleClass"], sortAscending: true });
                    $(this).html("Sort: A to Z");
                    this.setAttribute("data-filter", "ZA");
                } else if (filterName == "ZA") {
                    $grid.isotope({ sortBy: params["searchTitleClass"], sortAscending: false });
                    $(this).html("Sort: Z to A");
                    this.setAttribute("data-filter", "AZ");
                }
                updateLazyLoad();
                checkResults();
            });
        }

        //UPDATE LAZY LOADED IMAGES
        function updateLazyLoad() {
            setTimeout(function () {
                $("img.lazy").filter(function () {
                    var rect = this.getBoundingClientRect();
                    return rect.top >= 0 && rect.top <= window.innerHeight;
                }).trigger("lazylazy");
            }, 0750);
        }
    }

    // #region Lazy Load Images
    // On document ready start lazy load and isotope integration
    jQuery(document).ready(function () {
            var $win = $(window),
                $imgs = $("img.lazy"),
                $grid = "";


            // Grab isotope grid container
            if ($(".ac-listing-items").length > 0) {
                $grid = $(".ac-listing-items").isotope();
            }

            if ($(".nt-blog-listing-items").length > 0) {
                $grid = $(".nt-blog-listing-items").isotope();
            }

            if ($(".profile-listing-items").length > 0) {
                $grid = $(".profile-listing-items").isotope();
            }

            if ($(".honoree-listing-items").length > 0) {
                $grid = $(".honoree-listing-items").isotope();
            }

            // Load the images visible to the browser
            function loadVisible($els, trigger) {
                $els.filter(function () {
                    var rect = this.getBoundingClientRect();
                    return rect.top >= 0 && rect.top <= window.innerHeight;
                }).trigger(trigger);
            }

            // Once layout is complete load the visible images
            if ($(".profile-listing-items").length > 0) {
                $grid.isotope("on", "layoutComplete", function () {
                    loadVisible($imgs, "lazylazy");
                });
            }

            if ($(".honoree-listing-items").length > 0) {
                $grid.isotope("on", "layoutComplete", function () {
                    loadVisible($imgs, "lazylazy");
                });
            }

            // When the page is scrolled keep loading images
            $win.on("scroll", function () {
                loadVisible($imgs, "lazylazy");
            });

            // Lazy load image style
            $imgs.lazyload({
                effect: "fadeIn",
                placeholder: "/Assets/Images/default-place-holder-image.jpg",
                failure_limit: Math.max($imgs.length - 1, 0),
                event: "lazylazy"
            });

            setTimeout(function () {
                runIsotopeChecks();
            }, 0050);

            //****************************
            // Isotope Load more button
            //****************************
            if ($(".nt-blog-listing-items").length > 0) {
                var initShow = 12; //number of items loaded on init & onclick load more button
                var counter = initShow; //counter for load more button
                var iso = $grid.data('isotope'); // get Isotope instance

                loadMore(initShow); //execute function onload

                function loadMore(toShow) {
                    $grid.find(".hide").removeClass("hide");

                    var hiddenElems = iso.filteredItems.slice(toShow, iso.filteredItems.length).map(function (item) {
                        return item.element;
                    });
                    $(hiddenElems).addClass('hide');
                    $grid.isotope('layout');

                    //when no more to load, hide show more button
                    if (hiddenElems.length == 0) {
                        jQuery("#load-more").hide();
                    } else {
                        jQuery("#load-more").show();
                    };

                }

                //append load more button
                    $grid.after('<button id="load-more" class="feed-button"> Load More</button>');

                //when load more button clicked
                $("#load-more").click(function () {
                    if ($('#filters').data('clicked')) {
                        //when filter button clicked, set initial value for counter
                        counter = initShow;
                        $('#filters').data('clicked', false);
                    } else {
                        counter = counter;
                    };

                    counter = counter + initShow;

                    loadMore(counter);
                });

                //when filter button clicked
                $("#filters").click(function () {
                    $(this).data('clicked', true);

                    loadMore(initShow);
                });
            }
    });
