   async function getPopularResearch() {
            // Switched to Trending Research uri
            const url = '/api/users/trending-research/';
            const response = await fetch(url);
            return await response.json();
        }

        function trendingResearchItem(product) {
            let image = `
                    <source srcset="${product.image}" type="image/jpeg">
                    <img class="product-image" src="${product.image}" alt="${product.title}">
            `;

            if (product.webp) {
                image = `<source srcset="${product.webp}" type="image/webp">` + image;
            }

            const item = `<div class="col-md-4 featured-child">
                                <a style="text-align: left" href="${product.url}" role="button">
                                        <picture class="product-image card-img-top">
                                            ${image}
                                        </picture>
                                    <h5>${product.title} <i class="fas fa-arrow-circle-right " role="button" ></i></h5>
                                    <b>${product.services_title}</b>
                                </a>
                                <div class="featured-child-summary" >
                                    ${product.summary}
                                </div>
                            </div>
                            `
            return item;
        }

        async function renderTrendingResearchList() {
            const data = await getPopularResearch();
            const list = data.map(product => trendingResearchItem(product));
            const html = list.join('');
            if (html) {
                document.getElementById('trending-research').innerHTML = html;
            }
        }
        renderTrendingResearchList();
