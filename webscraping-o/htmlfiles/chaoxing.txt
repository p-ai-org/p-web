	function checkMobile() {	
			var str = $("#phone").val();
			if(!str){
				$("#phone_red").hide();
				$("#div_phone").removeClass("errorTip");
				$(".het364").getNiceScroll().resize();
				return true;
			}
			var re = /^1\d{10}$/;
			var re2 = /^(([0\+]\d{2,3}-)?(0\d{2,3})-)(\d{7,8})(-(\d{3,}))?$/;	
			var re3 = /^(([0\+]\d{2,3})?(0\d{2,3}))(\d{7,8})(-(\d{3,}))?$/;

				$("#content").getNiceScroll().resize();

				$("#div_phone").addClass("errorTip");
				$(".het364").getNiceScroll().resize();
				return false;
			}
		}



	function checkEmail(){
