  function createElementChangeListener(selector, callback) {
    const initializedNodes = new Set([]);

    document.querySelectorAll(selector).forEach(node => {
        callback(node);
        initializedNodes.add(node);
    });

    const mutationObserver = new MutationObserver(() => {
        const nodes = document.querySelectorAll(selector);
        nodes.forEach(node => {
            if (!initializedNodes.has(node)) {
                callback(node);
                initializedNodes.add(node);
            }
        });
    });

    mutationObserver.observe(document.body, {
        childList: true,
        subtree: true
    });
}

   function isElementInViewport(el, extraOffsetTop, extraOffsetBottom) {
      var extraOffsetTop = extraOffsetTop || 0;
      var extraOffsetBottom = extraOffsetBottom || 0;
      var top = el.offsetTop;
      var height = el.offsetHeight;
      while(el.offsetParent) {
        el = el.offsetParent;
        top += el.offsetTop;
      }
      var topEdge = top - extraOffsetTop;
      var bottomEdge = top + height + extraOffsetBottom;
      var topScreenEdge = window.pageYOffset;
      var bottomScreenEdge = window.pageYOffset + window.innerHeight;
      return (topEdge < bottomScreenEdge && topEdge > topScreenEdge) ||  // top edge in screen
        (bottomEdge < bottomScreenEdge && bottomEdge > topScreenEdge) || // bottom edge in screen
        (topEdge < topScreenEdge && bottomEdge > bottomScreenEdge); // belly in screen
    }

                  <div class="widget__brief">
                   <p>
                    One of the things about the pandemic we'll always remember are the people who got caught on camera completely losing it because they had to wear a mask inside of a supermarket or restaurant.
                    <br/>
                   </p>
                   <p>
                    These are the people who were given the opportunity to be either part of the problem or the solution to the global pandemic and they often unapologetically chose the former.
                   </p>
                   <p>
                   </p>
                   <div id="upworthyFreeStarVideoAdContainer">
                    <div id="freestar-video-parent">
                     <div id="freestar-video-child">
                     </div>
                    </div>
                   </div>
                   <p>
                    While some of these incidents result in ugly confrontations, a business owner in North Texas is having a little fun with the anti-maskers that frequent his restaurant. He's posted a sign notifying them, upfront, that they will be charged if they don't comply with his mark rule.
                   </p>

                 <div class="body-description">
                  <div class="widget__brief">
                   <p>
                    For Festus Oyinwola, a 19-year-old first-generation college student from Dallas, Texas, the financial burden of attending college made his higher education dreams feel like a faraway goal.
                   </p>
                   <p>
                    As his high school graduation neared, Oyinwola feared he would have to interrupt his educational pursuits for at least a year to save up to attend college.
                   </p>
                   <p>
                    That changed when Oyinwola learned of the
                    <a href="http://dallascountypromise.org/" rel="noopener noreferrer" target="_blank">
                     Dallas County Promise
                    </a>
                    , a new program launched by
                    <a href="https://commitpartnership.org/" rel="noopener noreferrer" target="_blank">
                     The Commit Partnership
                    </a>
                    , a community navigator that works to ensure that all North Texas students receive an equitable education.
                   </p>
                   <p>
                   </p>
                   <div id="upworthyFreeStarVideoAdContainer">
                    <div id="freestar-video-parent">
                     <div id="freestar-video-child">
                     </div>
                    </div>
                   </div>
                   <p class="shortcode-media shortcode-media-rebelmouse-image">
                    <img class="rm-shortcode" data-height="630" data-rm-shortcode-id="4f0cbe9852c7238d77a27060f8a57515" data-rm-shortcode-name="rebelmouse-image" data-runner-src="https://assets.rebelmouse.io/eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpbWFnZSI6Imh0dHBzOi8vYXNzZXRzLnJibC5tcy8yNTc0NzI3MC9vcmlnaW4uanBnIiwiZXhwaXJlc19hdCI6MTY3NzE2NDY0M30.QcoGN7qmh2v0A4klSvzXpOPpVyP3ZkNsgeG0JJ-5lWk/img.jpg?width=980" data-width="1200" id="0c4f5" type="lazy-image"/>
                    <small class="image-media media-photo-credit" placeholder="Add Photo Credit...">
                     Photo by
                     <a href="https://unsplash.com/@element5digital?utm_source=unsplash&amp;utm_medium=referral&amp;utm_content=creditCopyText" rel="noopener noreferrer" target="_blank">
                      Element5 Digital
                     </a>
                     on Unsplash
                    </small>
                   </p>
                   <p>
                    The Dallas County Promise covers any cost of tuition not included in financial aid grants. To date, nearly 60
                    <a href="http://dallascountypromise.org/students/highschools" target="_blank">
                     high schools
                    </a>
                    in Dallas County currently participate in this initiative.
                    <br/>
                   </p>
                   <p>
                    It pairs students  including Oyinwola  with a success coach for the following three years of their education.
                   </p>
                   <p>
                    To ensure that students like Oyinwola have the opportunity to build a solid foundation, The Commit Partnership is supported by businesses like Capital One who are committed to driving meaningful change in Dallas County through improved access to education.
                   </p>
                   <p>
                    The bank's support comes as part of its initial $200 million, multi-year commitment to advance socioeconomic mobility through the
                    <a href="https://www.capitalone.com/about/newsroom/impact-initiative-announcement/" target="_blank">
                     Capital One Impact Initiative
                    </a>
                    .
                   </p>

function intersectorCallback(video_player) {
	var booted = false;
	var totalSeconds = 0;
	var interval = null;

	function impressionCounter() {
		totalSeconds++;
		console.log("I've been watching for " + totalSeconds + " seconds");
	}
	return function intersector(entries) {
		var entry = entries[0];
		if (entry.isIntersecting) {
			if (!booted) {
				booted = true;
				video_player.mute();
				video_player.play();
			}
			interval = setInterval(impressionCounter, 1000);
		} else {
			if (interval) {
				console.log("I've stopped watching at " + totalSeconds + " seconds");
				clearInterval(interval);
				interval = null;
			}
		}
	};
}
