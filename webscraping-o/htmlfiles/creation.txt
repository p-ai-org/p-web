     (function() {
      var sliderEl = document.getElementById('cmi-slider-7824edc86d0973dbaa29f33bfb5316c0')
      // Slide controller.
      var slides = [].slice.call(sliderEl.querySelectorAll('.slide'))
      var currentSlideIndex = 0
      var currentTimeout

      function changeSlide(direction) {
        currentSlideIndex += direction

        // Reset to 0 if the target slide doesn't exist and moving forward
        // or reset to end of slide list if moving backwards.
        if (!slides[currentSlideIndex]) {
          currentSlideIndex = direction === -1 ? slides.length - 1 : 0
          return changeSlide(0)
        }

        var targetSlide = slides[currentSlideIndex]

        // Hide current active slide.
        ;(sliderEl.querySelector('.slide.show-slide') || {}).className = 'slide'
        targetSlide.className = 'slide show-slide'

        // Reschedule the time for the next slide change.
        clearTimeout(currentTimeout)
        currentTimeout = setTimeout(function() {
          changeSlide(1)
        }, targetSlide.dataset['delay'] || 6000)
      }

      sliderEl.querySelector('._next_button')
      .addEventListener('click', function() {
        changeSlide(1)
      })

      sliderEl.querySelector('._prev_button')
      .addEventListener('click', function() {
        changeSlide(-1)
      })

      changeSlide(0)
    })()

  // Load countries list dynamically to avoid excessive DOM size on page load.
  function populateCountries() {
    const cmi_country = 'us'

    return fetch('/plugins/system/geoipcountry/countryList.php')
      .then(res => res.json())
      .then(countries => {
        const core_countries = ['AU', 'NZ', 'CA', 'US','GB', 'SG', 'ZA']

        const other_countries = Object.keys(countries)
          .filter(key => core_countries.indexOf(key) === -1)
          .sort()

        const country_options = [].concat(
          // Add core country options.
          core_countries.map(code => {
            const name = countries[code]
            const selected = cmi_country.toUpperCase() === code
            return `<option value="${code}" ${selected ? 'selected' : ''}>${name}</option>`
          }),
          // Add "Other" pseudo-option
          ['<option value="--" disabled>-- Other --</option>`'],
          // Add non-core countries.
          other_countries.map(code => {
            const name = countries[code]
            const selected = cmi_country.toUpperCase() === code
            return `<option value="${code}" ${selected ? 'selected' : ''}>${name}</option>`
          })
        )

        document.getElementById('infobyte-signup__country').innerHTML = country_options.join('')
      })
  }

      /** Load fallback images from a comma-separated list of URLs in the data-fallback-src attribute on img elements. */
      var handleError = function() {
        // Get all fallback URLs from data attribute.
        var fallbackSrcs = this.getAttribute("data-fallback-src")
          .split(",").map(function(s) { return s.trim() })

        // Try the next URL if the current URL has already been tried.
        var currentSrc = fallbackSrcs.indexOf(this.src)
        if (currentSrc + 1 < fallbackSrcs.length) this.src = fallbackSrcs[currentSrc + 1]
      }

      // Attach an onerror handler
      var attachElements = document.querySelectorAll("[data-fallback-src]")
      for (var i = 0; i < attachElements.length; i++) {
        attachElements[i].addEventListener('error', handleError)
      }

      /** Hide images that failed to load */
      var imgElements = document.querySelectorAll('img')

          this.style.visibility = 'hidden';
        })

          this.style.visibility = 'visible';

  <script type="text/javascript">
   window.cmi_country_selector = {
	is_loaded: false,
	cookie: {
		cmi_country: "us",
		cmi_language: "en"	},

	populateCountries() {
		return fetch('/plugins/system/geoipcountry/countryList.php')
			.then(res => res.json())
			.then(countries => {
				const core_countries = ['AU', 'NZ', 'CA', 'US','GB', 'SG', 'ZA']

				const other_countries = Object.keys(countries)
					.filter(key => core_countries.indexOf(key) === -1)
					.sort()

				const country_options = [].concat(
					// Add core country options.
					core_countries.map(code => {
						const name = countries[code]
						const selected = this.cookie.cmi_country.toUpperCase() === code
						return `<option value="${code}" ${selected ? 'selected' : ''}>${name}</option>`
					}),
					// Add "Other" pseudo-option
					['<option value="--" disabled>-- Other --</option>`'],
					// Add non-core countries.
					other_countries.map(code => {
						const name = countries[code]
						const selected = this.cookie.cmi_country.toUpperCase() === code
						return `<option value="${code}" ${selected ? 'selected' : ''}>${name}</option>`
					})
				)

				document.getElementById('changeCountryVal').innerHTML = country_options.join('')
			})
	},

	populateLanguages() {
		return fetch('/plugins/system/geoipcountry/languageList.php')
			.then(res => res.json())
			.then(languages => {
				const language_options = languages.map(l => {
					const selected = this.cookie.cmi_language.toLowerCase() === l.code.toLowerCase()
					return `<option value="${l.code}" ${selected ? 'selected' : ''}>${l.name}</option>`
				})

				document.getElementById('changeLanguageVal').innerHTML = language_options.join('')
			})
	},

	changeCountry() {
		$('#changeCountry').modal('show')

		if (!this.is_loaded) {
			this.toggleLoader(true)

			// Load country and language lists.
			Promise.all([this.populateCountries(), this.populateLanguages()])
			.then(() => {
				// Load the google api for finding locations.
				$.getScript("https://maps.googleapis.com/maps/api/js?key=AIzaSyCdGbnd5cepgzMjLUW42uGUfxQ4Y2Mqty8&libraries=places", () => {
					 this.swapCountryValue(false)
				});
				this.is_loaded = true
				this.toggleLoader(false)
			})
		} else {
			this.swapCountryValue(false)
		}
	},

	toggleLoader(state) {
		if (state) {
			$('#changeCountryLoader').show()
			// $('#changeCountryForm').hide()
		} else {
			$('#changeCountryLoader').hide()
			// $('#changeCountryForm').show()
		}
	},

	swapCountryValue(wipe) {
		this.autoCompleteChangePostcode()

		var eventOffices = ["UK", "GB", "AU", "CA", "NZ", "ZA", "SG", "US"]
		var oMatch = eventOffices.indexOf($('#changeCountryVal').val())
		if (oMatch >= 0){
			$('#askLocation').show()

		} else {
			if (wipe == true){
			 	$('#changeLocationPostcode').val('')
			 	$('#changeLocationLat').val('')
			 	$('#changeLocationLng').val('')
			 	$('#changeLocationName').val('')
			 	$('#changeLocationCity').val('')
			 	$('#changePostcode').val('')
			}
			$('#askLocation').hide()
		}

		if (wipe == true){
		 	$('#changeLocationPostcode').val('')
		 	$('#changeLocationLat').val('')
		 	$('#changeLocationLng').val('')
		 	$('#changeLocationName').val('')
		 	$('#changeLocationCity').val('')
		 	$('#changePostcode').val('')
		}
	},

	autoCompleteChangePostcode() {
		var setoffice = $('#changeCountryVal').val()
		autocomplete = new google.maps.places.Autocomplete(
			document.getElementById('changePostcode'),
			{types: ['(regions)'], componentRestrictions: {country:setoffice.toUpperCase()}}
		)

	  autocomplete.addListener('place_changed', () => {
			this.changeAutoCompleteSelected(autocomplete)
		})
	},

	changeAutoCompleteSelected(autocomplete) {
		var place = autocomplete.getPlace()
		if (place.geometry == undefined) {
				jQuery('#nosearchfound').show()
				return false
		}

	  $('#changeLocationCity').val(place.name)
	  $('#changePostcode').val(place.formatted_address)

	  $('#changeLocationLat').val(place.geometry.location.lat())
	  $('#changeLocationLng').val(place.geometry.location.lng())
	  if (place.vicinity != undefined){
		  $('#changeLocationName').val(place.vicinity)
	   }
		var namefound = 0;

		for (var i = 0; i < place.address_components.length; i++) {
			for (var j = 0; j < place.address_components[i].types.length; j++) {
				if (place.address_components[i].types[j] == "locality") {
						$('#changeLocationName').val(place.address_components[i].long_name)
						namefound = 1
				}
			}
		}

	  if (namefound == 0 && place.vicinity != undefined) {
		  $('#changeLocationName').val(place.vicinity)
	  }
	},

	getBackupPostcodeForOffice(office) {
		return {
			au: '4078',
			ca: 'N2M 5G8',
			nz: '2145',
			sg: '911207',
			za: '7551',
			uk: 'LE8 6JS',
			gb: 'LE8 6JS',
			us: '30127'
		}[office] || '4078'
	},

	changeCountrySubmit() {
		this.toggleLoader(true)
		var changeLanguageVal = $("#changeLanguageVal").val()
		var changeCountryVal = $("#changeCountryVal").val()
		var changeCity = $('#changeLocationCity').val()
		var changePostcode = $('#changeLocationPostcode').val()
		if (!changePostcode) changePostcode = this.getBackupPostcodeForOffice(changeCountryVal)
		var lat = $('#changeLocationLat').val()
		var lng = $('#changeLocationLng').val()
		var location_name = $('#changeLocationName').val()

		$.post('', { changeCountry: changeCountryVal, changeLanguage: changeLanguageVal, lat:lat, lng:lng, location_name:location_name }, function(data) {
			window.location.reload()
		})
	}
}
