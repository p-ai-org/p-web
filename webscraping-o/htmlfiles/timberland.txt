function JsEqualHeight()
{

	this.resolution = undefined;

	this.getResolution = function()
	{
		// Small
		if (window.innerWidth <= 639)
			return "for-small";

		// Medium
		if (window.innerWidth >= 640 && window.innerWidth <= 1023) 
			return "for-medium";

		// Large
		if (window.innerWidth >= 1024) 
			return "for-large";
	}

	this.equalizeHeight = function(nodes, jsAttribute)
	{
		var attributes = [];
		for (var i = 0 ; i <= nodes.length-1 ; i++)
		{
			var attribute = nodes[i].getAttribute(jsAttribute);
			if (attributes.indexOf(attribute) == -1)
			{
				attributes.push(attribute)
				var nodesByAttribute = document.querySelectorAll('[' + jsAttribute + '="' + attribute + '"]');
				var h = 0;
				var w = 0;
				var _ii = 0;
				for (var ii = 0 ; ii <= nodesByAttribute.length-1 ; ii++)
				{
					nodesByAttribute[ii].style.height = 'auto';
					w += nodesByAttribute[ii].clientWidth;
					if (ii == nodesByAttribute.length-1)
					{
						h = Math.max(nodesByAttribute[ii].clientHeight, h);
						for (var iii = _ii ; iii <= ii ; iii++)
							nodesByAttribute[iii].style.height = h + 'px';
					}
					else
					{
						if (w > window.innerWidth || ii == nodesByAttribute.length-1)
						{
							for (var iii = _ii ; iii <= ii-1 ; iii++)
								nodesByAttribute[iii].style.height = h + 'px';
							h = nodesByAttribute[ii].clientHeight;
							w = nodesByAttribute[ii].clientWidth;
							_ii = ii;
						}
						h = Math.max(nodesByAttribute[ii].clientHeight, h)
					}
				}
			}
		}
	}

	this.resize = function()
	{ 
		this.resolution = this.getResolution();
		var nodes = document.querySelectorAll('[js-equal-height-for-small], [js-equal-height-for-medium], [js-equal-height-for-large]');
		for (var ii = 0 ; ii <= nodes.length-1 ; ii++)
		{
			if (nodes[ii].style.height)
				nodes[ii].style.height = 'auto';
		}
		var jsEqualHeightFor = document.querySelectorAll('[js-equal-height-' + this.resolution + ']');
		var jsEqualHeight = document.querySelectorAll('[js-equal-height]');
		this.equalizeHeight(jsEqualHeightFor , 'js-equal-height-' + this.resolution);
		this.equalizeHeight(jsEqualHeight , 'js-equal-height');

		window.requestAnimationFrame(this.resize.bind(this));
	}

	window.requestAnimationFrame(this.resize.bind(this));
}

function Animate()
{
	this.elements = [];
	this.requestAnimationFrameCallBack = undefined;

	this.resolution = undefined;
	this.resize = function()
	{
		// Small
		if (window.innerWidth <= 639)
			this.resolution =  "small";

		// Medium
		if (window.innerWidth >= 640 && window.innerWidth <= 1023) 
			this.resolution = "medium";

		// Large
		if (window.innerWidth >= 1024) 
			this.resolution = "large";
	}

	this.parsePage = function()
	{
		this.elements = document.querySelectorAll('.animate');
	}
	this.requestAnimationFrame = function()
	{			
		if (this.requestAnimationFrameCallBack)
			this.requestAnimationFrameCallBack();
		var scrollY = Math.max(window.pageYOffset, document.documentElement.scrollTop, document.body.scrollTop);
		var screenMiddleHeight = window.innerHeight/2;

		for (var i = 0 ; i <= this.elements.length-1 ; i++)
		{
			var elementMiddleY = this.elements[i].getBoundingClientRect().top + (this.elements[i].clientHeight/2);
			var animate = JSON.parse(this.elements[i].getAttribute('animate'));
			for (var ii = 0 ; ii <= animate.length-1 ; ii++)
			{
				if (animate[ii].animateFor.indexOf(this.resolution) != -1)
				{
					var y = (window.innerHeight/100)*animate[ii].y;
					var deltaMiddleY = (y - elementMiddleY);
					if (deltaMiddleY >= 0 && animate[ii].infinite == 'false')
						value = animate[ii].to;
					else
					{	
						value  = (((  animate[ii].from - animate[ii].to) * deltaMiddleY) / (window.innerHeight - y));
						value = animate[ii].to -value;
					}
					this.elements[i].style[animate[ii].cssProperty] = animate[ii].cssValue.replace('#', value);	
				}
				else
				{
					this.elements[i].style[animate[ii].cssProperty] = '';	
				}						
			}
		}
		window.requestAnimationFrame(this.requestAnimationFrame.bind(this));
		window.addEventListener('resize', this.resize.bind(this));
	}

	this.main  = function()
	{
		this.parsePage();
		this.resize();
		window.requestAnimationFrame(this.requestAnimationFrame.bind(this));
	}
	this.main();
}
