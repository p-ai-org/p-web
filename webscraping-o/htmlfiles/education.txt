                        return ('<p data-term="' + data.term + '" data-display="' + data.display + '">' + (data.display).replace(data._query, '<strong>' + data._query + '</strong>') + '</p>');
                    }
                }

                if(e.which === 13) {

                    var termFound = false;

                        if (typeAheadInput.val() === $(el).data().display || typeAheadInput.val() === $(el).data().term) {
                            var searchTerm = decodeURIComponent($(el).data().term)
                                .replace(/[/+]/g,' ')
                                .replace(/(^\s+|\s+$)/g,'');
                            typeAheadInput.val(searchTerm);
                            // submitSearchbButton.click();
                            form.submit();
                        }
                    }).promise().done( function() {
                        var searchTerm = decodeURIComponent(typeAheadInput.val())
                            .replace(/[/+]/g,' ')
                            .replace(/(^\s+|\s+$)/g,'');
                        typeAheadInput.val(searchTerm);
                        // submitSearchbButton.click();
                        form.submit();
                    });
                }
            });

            typeAheadInput.bind('typeahead:select', function(e, suggestion) {
                var searchTerm = decodeURIComponent(suggestion.term)
                    .replace(/[/+]/g,' ')
                    .replace(/(^\s+|\s+$)/g,'');

                    .replace(/[/+]/g,' ')
                    .replace(/(^\s+|\s+$)/g,'');
                typeAheadInput.val(searchTerm);
                form.submit();
            });
        };

        if($.fn.typeahead) {
            onTypeaheadLoad();
        }
        else {
            $(document).on('typeaheadLoaded', onTypeaheadLoad);
        }
