         <h5>
          <span>
           What is the difference between Shoutcast For Business &amp; Shoutcast Server Software ?
          </span>
         </h5>
         <div id="faq-1">
          <span>
           Shoutcast Server Software is a software that is installed on your own network server,
                                while with Shoutcast for business, we host your stations on our servers, all you need to broadcast is an internet access.
           <br/>
           For both these products, you can monetize your radio station and access the Radio Manager Online.
          </span>

           Do you offer a free trial?
          </span>

            function checkCountryCookie() {
                var cookie = getCookie("usercountry");
                if (cookie == null || cookie == "null" ||cookie == "") {
                    var client = new XMLHttpRequest();
                    client.onload = handleCountryCall;
                    client.open("GET", "Home/GetCountry");
                    client.send();
                }
            }

            function handleCountryCall() {
                if (this.status == 200) {
                    var response = JSON.parse(this.responseText);
                    if (response == null) {
                        setCookie("usercountry", null);
                        setCookie("usercurrency", "Dollars");
                        checkCurrencyCookie();
                    } else {
                        setCookie("usercountry", response.name);
                        if (response.continentCode == "EU") {
                            setCookie("usercurrency", "Euros");
                            checkCurrencyCookie();
                        } else {
                            setCookie("usercurrency", "Dollars");
                            checkCurrencyCookie();
                        }
                    }
                } else {
                    setCookie("usercountry", null);
                    setCookie("usercurrency", "Dollars");
                    checkCurrencyCookie();
                }
            }
