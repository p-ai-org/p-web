        // Format Alerts
        function formatUtmbAlerts(utmbAlertsViewModel) {

            // Get Alert Modal Css ClassName
            function getAlertModalClassName(catThreatLevel) {
                var result = '';
                if (catThreatLevel) {
                    switch (catThreatLevel.toLowerCase()) {
                        case "red":
                            result = 'msg-red-alert';
                            break;
                        case "yellow":
                            result = 'msg-yellow-alert';
                            break;
                        case "orange":
                            result = 'msg-orange-alert';
                            break;
                        case "green":
                            result = 'msg-green-alert';
                            break;
                    }
                }
                return result;
            }

            // Get Alert Messages
            function getAlertMessages(alerts) {
                var result = '';
                if (alerts) {
                    for (var i = 0; i < alerts.length; i++) {
                        result += '<div class="alert-msg-container">';
                        result += '<p class="alert-msg-title">' + alerts[i].AlertTitle + '</p>';
                        result += '<p class="alert-msg-meta">' + alerts[i].AlertPostDate + ' &bull; ' + alerts[i].AlertPostTime + ' &bull; ' + alerts[i].AlertUsername + '</p>';
                        result += '<div class="alert-msg">' + alerts[i].AlertMessage + '</div>';
                        //result += '<hr />';
                        result += '</div>';
                    }
                }
                return result;
            }

            // Alert Message Count
            function getAlertMessageCount(alertCount) {
                var result = '';
                if (alertCount > 1) {
                    result = ' <span class="badge alert-count-badge"><small>' + alertCount + '</small></span>';
                }
                return result;
            }

            // Extend this Thing
            // Add nested alert messages from category's 'UtmbAlertMessages' array
            return $.extend(utmbAlertsViewModel, {
                FormattedModalClassName: getAlertModalClassName(utmbAlertsViewModel.AlertCatThreatLevel),
                FormattedAlertMessages: getAlertMessages(utmbAlertsViewModel.UtmbAlertMessages),
                AlertMessageCount: getAlertMessageCount(utmbAlertsViewModel.AlertCount)
            });
        };


        // Get Entries
        function getAlerts() {

            // Empty Container
            //$utmbAlertsContainer.empty();
            $('#utmbAlertSystem').remove();
            
            // Prepend Alert Bar Control
            $('#PublicWrapper').prepend(getAlertBarControl());

            //Fetching UTMB Alerts
            return utmbAlertsHub.server.getUtmbAlerts().done(function (utmbAlerts) {
                // Append Rows
                $.each(utmbAlerts, function () {
                    var utmbAlertsViewModel = formatUtmbAlerts(this);
                    $('#utmbAlertSystem').find('#alertsList').append(rowTemplate.supplant(utmbAlertsViewModel));
                });
            });
        };


        // Set Alert Bar is-active ClassName
        function setActiveClass() {
            var alertBar = document.getElementById('utmbAlertSystem');
            if (alertBar) alertBar.className = 'is-active';
        };


        // [CRUD MANAGER] Manage this Thing
        utmbAlertsHub.client.UtmbAlertsManager = function (action, utmbAlertsViewModel) {
            var thisUtmbAlert = formatUtmbAlerts(utmbAlertsViewModel);
            var $thisUtmbAlertRow = $(rowTemplate.supplant(thisUtmbAlert));
            var $thisUtmbAlertId = $('.alert-' + thisUtmbAlert.AlertCatId);

            switch (action) {

                case 'createRecord':
                    console.log("A new alert was created.");
                    $utmbAlertsList.prepend($thisUtmbAlertRow);
                    //$thisUtmbAlertRow.flash(rowBg, 1000);
                    break;

                case 'viewRecord':
                    console.log("Alert is currently currently locked as it's being viewed.");
                    $thisUtmbAlertId.closest("li").replaceWith($thisUtmbAlertRow);
                    //$thisUtmbAlertRow.flash(rowBg, 1000);
                    break;

                case 'updateRecord':
                    console.log("An alert record was updated.");
                    $thisUtmbAlertId.closest("li").replaceWith($thisUtmbAlertRow);
                    //$thisUtmbAlertRow.flash(rowBg, 1000);
                    break;

                case 'deleteRecord':
                    $thisUtmbAlertId.closest("li").remove();
                    break;
            }
        };

        // [LOG: OPTIONAL] Browser Console Log
        $.connection.hub.logging = true;

        // [LOG ERRORS] Only for Dev
        // -http://stackoverflow.com/questions/19688673/signalr-there-was-an-error-invoking-hub-method-xxx
        $.connection.hub.error(function (error) {
            console.log('SignalR error: ' + error)
        });

        // Start the connection
        $.connection.hub.start({ transport: ['serverSentEvents', 'longPolling'] })
            .then(getAlerts)
            .then(setActiveClass);

    });

                           <!-- Description -->
                           <desc>
                            <p>
                             UTMB presents this conference as a platform for wide discussion surrounding four issues:
                            </p>
                            <ol>
                             <li>
                              Social, economic, and environmental determinants
                             </li>
                             <li>
                              Health care access, quality, and efficiency
                             </li>
                             <li>
                              Migration of health personnel (brain drain)
                             </li>
                             <li>
                              Migrant detention
                             </li>
                            </ol>
                            <p>
                             Migration is a determinant of health. Displacement and legal status expose people to vulnerability, deprivation, discrimination, violence, abuse, and extortion, and limit their access to healthcare, thus affecting both immediate and lifelong health outcomes. The COVID-19 pandemic has exacerbated existing conditions of vulnerability and social exclusion of marginalized populations, including migrants, especially unauthorized migrants, those in detention centers, refugees, and asylum seekers. Thus, they experience a disproportionately higher pandemic burden. The Annual Conference on the Health of Migrants provides a platform for constructive policy dialogue among relevant stakeholders, including immigrants and organizations representing migrants, law and policymakers, government leaders, health care professionals, the faith community, health system administrators, academicians, researchers, community-based service providers, and international and multilateral organizations.

                        more than 2,500 students and more than 1,000 faculty.
