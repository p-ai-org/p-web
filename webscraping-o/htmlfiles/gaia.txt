     $(document).ready(function () {

    var $primary_haspopup = $('.nav-primary__item--has-children');

    function handletoggleAriaExpanded(selector, reset) {
      $(selector).attr('aria-expanded', function (i, attr) {
        return (attr === 'true' || reset) ? 'false' : 'true'
      });
    }

    // Hides and resets nav items
    function reset_nav_items() {
      // Reset all to expanded false
      $primary_haspopup.each(function () {
        handletoggleAriaExpanded($(this).find('a').first(), true);
      });

      // Set all primary inactive
      $('.nav-primary__item').removeClass('nav-primary__item--active').find('a').removeClass('active');

      // Hide all secondary
      $('.nav-secondary').hide();

      // Hide autocomplete
      $('.autocomplete-suggestions').css('display', 'none');
    }

    // Hides Login/Signup for Logged in Users
    if ($.fn.isLoggedIn()) {
      $('#login_webapp_header, #signup_webapp_header').hide();
    }

    // Hide nav events on window resizing from mobile to desktop
    $(window).resize(function () {
      //if ($(this).width() > 1155) {
      $(document).trigger('resize.webappNavigationDropdownMenu');
      //}
    });

    // Handles the mobile nav toggle (hamburger menu)
    $('.navbar-toggler').on('click', function () {
      // update the hamburger icon
      $(this).toggleClass('open');
      // show the nav
      $('.webapp-header__nav').toggleClass('webapp-header__nav--open');

      if ($(this).hasClass('open')) {
        $('.content-wrapper').css({position: 'fixed', width: '100%'});
      } else {
        $('.content-wrapper').css({position: 'relative', width: 'auto'});
        // Set all primary to inactive
        $('.nav-primary__item').removeClass('nav-primary__item--active').find('a').removeClass('active');
        // Hide all secondary
        $('.nav-secondary').hide();
      }

    });

    // Special handling for desktop haspopup items - link out instead of display menu
    $('.nav-primary__item--mobile-no-link').on('click', function (e) {

      e.preventDefault();
      // only link out if not in mobile view
      if (!$('.navbar-toggler').hasClass('open')) {
        $.fn.linkCheck($(this));
      }

    });

    // Handles secondary menu display
    $primary_haspopup.on('click', function (e) {

      e.preventDefault();

      // only show on mobile menu view except for the ellipsis "more" menu on desktop
      if (!$('.navbar-toggler').hasClass('open') && !$(this).hasClass('nav-primary__item--more')) {
        return;
      }

      var target = $(this).find('.nav-secondary'),
          $this = $(this);
      if (target.css('display') === "none") {

        // sets default states to all items before setting current item active/open
        reset_nav_items();
        // set active state
        $this.addClass('nav-primary__item--active').find('a').first().addClass('active');
        // set expanded state
        handletoggleAriaExpanded($this.find('a').first());

        // Handle display show
        target.fadeIn(250, function () {

          if ($(this).css('display') === "block") {

            // add close listener once shown
            $(document).on('resize.webappNavigationDropdownMenu', function () {
              reset_nav_items();
              // Clear out listeners
              $(document).off('resize.webappNavigationDropdownMenu');
            });

            if ($(window).width() > 1155) {

              $(document).on('click', function () {
                reset_nav_items();
                // Clear out listeners
                $(document).off('click');
              });

            }

          }

        });
      }

    });

    // Language Selection once Clicked
    $('.webapp-header__language-item').on('click', function () {
      // Hide and reset drop down UI
      $('.webapp-header__language-select').hide();
      $('.webapp-header__toggle-down-select').show();
      // Set language preferences in local storage
      $.fn.setLanguagePreference($(this).attr('data-lang'));
    });

    $('.webapp-header__language-toggle').on('click', function () {

      var $this = $(this);

      handletoggleAriaExpanded($this);

      //Checks if language menu is showing
      var toggle_down = $('.webapp-header__toggle-down'),
          toggle_up = $('.webapp-header__toggle-up'),
          lang_select = $('.webapp-header__language-select');
      if (toggle_down.is(":visible")) {
        toggle_down.hide();
        toggle_up.show();
        lang_select.fadeIn(500, function () {
          // handles triangle elements
          $(document).on('click.webappNavigation', function () {
            $(document).off('click.webappNavigation');
            toggle_down.show();
            toggle_up.hide();
            lang_select.hide();

            handletoggleAriaExpanded($this);

          });
        });
      } else {
        toggle_down.show();
        toggle_up.hide();
        lang_select.hide();
      }
    });

    $("#webapp-search-input").autocomplete({
      serviceUrl: GAIA_GLOBALS.dbi_brooklyn_api + "/search/autocomplete?language=" + GAIA_GLOBALS.lang,

    <script>
     $(document).ready(function () {
    // handle displays
    $('.layer-popup').each(function () {
      // GA Tracking fires
      var target = $(this),
          ga_label = $(this).attr('data-pu-ga-event-label');

      switch ($(this).attr('data-pu-trig')) {
        case 'timer':
          var dur = $(this).attr('data-pu-delay'),
              pu_timeout = null;

          if (pu_timeout != null) { clearTimeout(pu_timeout); }
          if (dur > 0) {
            pu_timeout = setTimeout(function () {
              openTemplatePopup(target);
            }, dur * 1000);
          } else {
            openTemplatePopup(target);
          }
          break;
        case 'scroll':
          var visible = false;
          $(document).scroll(function () {
            // check again 75% of the full screen height
            if ($(window).scrollTop() > ($(window).innerHeight() * 0.75)) {
              if (!visible) {
                visible = true;
                openTemplatePopup(target);
              }
            } else {
              if (visible) {
                visible = false;
                if ($(window).width() <= 768) {
                  $(target).animate({
                    bottom: "-1000px",
                    right: "initial"
                  }, function () {
                    // Dispatch close event to children element dependencies
                    $(target).trigger('popup:rightSideClose');
                  });
                } else {
                  $(target).animate({
                    right: "-2000px",
                    bottom: "initial"
                  }, function () {
                    // Dispatch close event to children element dependencies
                    $(target).trigger('popup:rightSideClose');
                  });
                }
                if (ga_label) {
                  // Popup Closed/Hidden by scroll
                  //console.log("GA: Popup Hidden -> ", ga_label);
                  dataLayer.push({
                    'event': 'customEvent',
                    'eventCategory': 'pop-up',
                    'eventAction': 'exit click',
                    'eventLabel': ga_label // name of widget
                  });
                }
              }
            }
          });
          // Handle users that change screen size mid experience (low use case)
          $(window).resize(function () {
            visible = false;
            if ($(window).width() <= 768) {
              $(target)
                  .css({
                    top: "initial",
                    right: "initial",
                    bottom: "-1000px"
                  });
            } else {
              $(target)
                  .css({
                    top: "48%",
                    bottom: "initial",
                    right: "-2000px"
                  })
            }
          });
          break;
        case 'click':
          // reference the value set in CMS for btn target
          var btn_class = $(this).attr('data-pu-btn-listener');
          $(this).on("popup:clickOpenPopup", function (e, btn) {
            if ($(btn).hasClass(btn_class)) {
              openTemplatePopup(target);
            }
          });
          break;
        case 'exit':
          var shown = false;
          $(document).mouseleave(function (e) {
            if ((e.clientY < 2
                || (e.clientY < 100 && e.clientX > $(window).innerWidth())
                || (e.clientY < 100 && e.clientX <= 0))
                && shown === false) {
              openTemplatePopup(target);
              shown = true;
            }
          });
          break;
      }
      // CTA Buttons
      if (ga_label) {
        $(this).find('.layer-popup__cta').click(function () {
          //console.log("GA: CTA clicked -> ", ga_label);
          // Popup CTA Clicks
          dataLayer.push({
            'event': 'customEvent',
            'eventCategory': 'pop-up',
            'eventAction': 'CTA-click',
            'eventLabel': ga_label
          });
        });
      }
    });
    function closeCookieActions(cname) {
      // If Cookies enabled, set accordingly
      if (cname !== undefined) {
        if ($.fn.getCookieAllowance()) {
          //console.log("cookies allowed -> ", cname)
          Cookies.set(cname, "I", {expires: 30});
        } else {
          //console.log("cookies NOT allowed")
        }
      } else {
        //console.log("No Cookies, move on..........")
      }
    }
    function openTemplatePopup(target) {
      var pos = $(target).attr('data-pu-pos'),
          trigger = $(target).attr('data-pu-trig'),
          ga_label = $(target).attr('data-pu-ga-event-label'),
          show_logged_in = $(target).attr('data-pu-showloggedin'),
          cname = $(target).attr('data-pu-cname');
      /*
       --Make sure we are allowed to show popup--
       Don't show to logged in user unless override is enabled
       Don't show to cookied users if cookies enabled
       */
      if (($.fn.isLoggedIn() && !show_logged_in) || Cookies.get(cname) === 'I') {
        return;
      }
      switch (pos) {
        case 'overlay':
          $(target)
              .fadeIn()
              .find('.close-popup-btn').click(function () {
            $(target).hide();
            closeCookieActions(cname);
          });
          break;
        case 'ribbon':
          $(target)
              .slideDown()
              .find('.close-popup-btn').click(function () {
            $(target).hide();
            closeCookieActions(cname);
          });
          break;
        case 'top':
          $(target)
              .slideDown()
              .find('.close-popup-btn').click(function () {
            $(target).slideUp();
            closeCookieActions(cname);
          });
          break;
        case 'right':
          // Mobile/Tablet (sm) slide up from bottom
          if ($(window).width() <= 768) {
            $(target)
                .animate({
                  bottom: "0px"
                }, function () {
                  // Dispatch open event to children element dependencies
                  $(target).trigger('popup:rightSideOpen');
                });
            // Desktop slide in from side
          } else {
            $(target)
                .animate({
                  right: "0px"
                }, function () {
                  // Dispatch open event to children element dependencies
                  $(target).trigger('popup:rightSideOpen');
                });
          }
          break;
        case 'bottom':
          $(target)
              .animate({
                bottom: "0px"
              })
              .find('.close-popup-btn').click(function () {
            $(target).animate({
              bottom: "-2000px"
            });
            closeCookieActions(cname);
          });
          break;
      }
      // GA open and close events
      if (ga_label) {
        //console.log("popup Trigger: ", trigger);
        // Popup Displays
        switch (trigger) {
          case 'timer':
            break;
          case 'scroll':
            /*dataLayer.push({
                'event': 'customEvent',
                'eventCategory': 'pop-up',
                'eventAction': 'displayed',
                'eventLabel': 'side scroller'
            });*/
            break;
          case 'click':
            /*dataLayer.push({
                'event': 'customEvent',
                'eventCategory': 'pop-up',
                'eventAction': 'displayed',
                'eventLabel': 'cta button'
            });*/
            break;
          case 'exit':
            /*dataLayer.push({
                'event': 'customEvent',
                'eventCategory': 'pop-up',
                'eventAction': 'displayed',
                'eventLabel': 'page abandonment'
            });*/
            break;
        }
        //console.log("GA: Popup Displays -> ", ga_label);
        dataLayer.push({
          'event': 'customEvent',
          'eventCategory': 'pop-up',
          'eventAction': 'displayed',
          'eventLabel': ga_label
        });
        $(target)
        // Popups that have close buttons
        .find('.close-popup-btn').click(function () {
          // Popup Closed
          //console.log("GA: Popup Closed -> ", ga_label);
          dataLayer.push({
            'event': 'customEvent',
            'eventCategory': 'pop-up',
            'eventAction': 'exit click',
            'eventLabel': ga_label
          });
        })
        // on popups that contain forms
        .find('form').submit(function () {
          // Email Collected (Pop-Up)
          //console.log("GA: Popup Form Submit -> ", ga_label);
          dataLayer.push({
            'event': 'customEvent',
            'eventCategory': 'popup',
            'eventAction': 'email submitted',
            'eventLabel': ga_label
          });
        });
      }
    }
  });
