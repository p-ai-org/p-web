        function ConvertToJsonParsableString(str) {
            try {
                var str_orig = str;
                str = htmlDecode(str);
                str = str.replace('"', "");     // then take off any terminal "
                str = ReverseString(str);
                str = str.replace('"', "");
                str = ReverseString(str);
