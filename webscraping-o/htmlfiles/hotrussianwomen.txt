        let $cur_btn = null;

        function menu_item_click(e) {
            e.preventDefault();

            if ($button.hasClass('open')) {
                $button.removeClass('open');
                $menu_header.removeClass('open');
            } else {

        function menu_open_sub_menu(e) {
            e.preventDefault();
            e.stopPropagation();

            let btn_click = $(this);

            if (!btn_click.is($cur_btn)) {
                let btn_active = $(btn_click).parent('li').parent('ul');
                btn_active.find('li.parent').find('a , span').removeClass('active');
                btn_active.find('li > ul.dropdown-menu.open').slideUp().removeClass('open');

            $(this).next('ul').toggle("slow", function () {
                if ($(this).hasClass('open')) {
                    $(this).removeClass('open');
                    btn_click.removeClass('active');
                } else {
                    $(this).addClass('open');
                    btn_click.addClass('active');
                }
            });

            $cur_btn = btn_click;
        }

        function menu_resize_version(){
            if ($(window).width() > 1000 && !pc_version) { //PC

                pc_vers = false;

                mob_vers = false;
                pc_vers = true;

                if ($(window).width() < 620) { //мобильные

                        mob_vers = true;
                        pc_vers = false;
                    }
                } else {
                    if (!pc_vers) {
                        hide_text_for_pc();
                        mob_vers = false;
                        pc_vers = true;
                    }
                }
