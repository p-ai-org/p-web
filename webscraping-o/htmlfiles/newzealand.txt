     *
     * @param {string} cname
     * @param {string} cvalue
     * @param {number} exdays
     * @param {string} cdomain
     */
    pub.set = function(cname, cvalue, exdays,cdomain) {
        var d = new Date();
        var cookiedomain = (!!cdomain)?cdomain:_data.defaultDomain;
        var expires = '';
        var secure = ';secure';
        if(exdays && exdays > 0){
            //If we have a day time frame use it, else we're going session cookies
            d.setTime(d.getTime() + (exdays*24*60*60*1000));
            expires = "expires="+d.toUTCString();
        }
        //var secure = ''; //Insecure cookies till site is fully https
        document.cookie = cname + "=" + cvalue + "; " + expires + secure + " ;domain="+cookiedomain+";path=/";
    };

    /**
     *
     * @param {string} cname
     */
    pub.get = function(cname) {
        var name = cname + "=";
        var ca = document.cookie.split(';');
        for(var i=0; i<ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0)==' ') c = c.substring(1);//Remove leading space
            if (c.indexOf(name) === 0) return c.substring(name.length, c.length); // MUST MATCH FROM START OF STRING.
        }
        return undefined;
    };
