var LOCATION_SERVICE = 'https://pro.ip-api.com/json/?fields=country,countryCode&key=orQz5rZZnd1GB44';

var ENGLISH_COUNTRIES = ['UK', 'US'];
var ALLOWED_DOMAINS = ['english.com'];
var BLOCKED_COUNTRIES = ['CN', 'RU', 'NOT'];
var PORTUGUESE_COUNTRIES = ['BR', 'PT'];
var SPANISH_COUNTRIES = ['AR', 'BO', 'BR', 'CL', 'CO', 'CR', 'CU', 'DO', 'EC', 'SV', 'GT', 'HN', 'MX', 'NI', 'PA', 'PY', 'PE', 'PR', 'ES', 'UY', 'VE'];
var IFRAME_PAGES = ['terms-of-use-spanish', 'terms-of-use-english', 'terms-of-use-portuguese'];
var SUBDOMAINS_RULES = [{
  subdomain: 'en',
  route: "//en.".concat(getHostname()),
  countries: ENGLISH_COUNTRIES
}, {
  subdomain: 'pt',
  route: "//pt.".concat(getHostname()),
  countries: PORTUGUESE_COUNTRIES
}, {
  subdomain: 'es',
  route: "//es.".concat(getHostname()),
  countries: SPANISH_COUNTRIES
}];

window.onload = function () {
  if (!ALLOWED_DOMAINS.includes(getHostname()) || IFRAME_PAGES.includes(getPathname())) {
    return showBody();
  }

  if ( getCookie('country') && getCookie('ipcountry') ) {
    validateBlockList();
    selectCountryFromCookie();
    validateLangDomain();
  } else {
    var load = function load(location) {
      var countryCode = location.countryCode || 'UK';
      setCookie('ipcountry', location.countryCode, 30);
      setCookie('country', Object.values(getCountries()).includes(countryCode) ? countryCode : 'UK', 30);
      validateBlockList();
      selectCountryFromCookie();
      validateLangDomain();
    };

    }).then(load, load);

  });

  if (getPathname() === '') {
    document.querySelector('#hero-form').addEventListener("submit", function (event) {
      onSubmitGetStarted(event);
    });
    
    document.addEventListener('click', function (event) {

    },false);

    showTeachWithUsForm();

    showIframe();
  }

function validateBlockList() {
  var countryCode = getCookie('ipcountry') || 'NOT';
  if(BLOCKED_COUNTRIES.includes(countryCode) && getPathname() !== '403'){
  	window.location.href = '/403';
  }
}

function onSubmitGetStarted(event) {
  event.preventDefault();
  var input = event.target.parentNode.querySelector('input[name="Email"]');
  var countryCode = getCookie('country') || 'UK';
  var endpoint = REDIRECT_SIGN_UP;
  window.location = "".concat(endpoint, "?email=").concat(input.value, "&country=").concat(countryCode);
}

function onSelectCountry(event) {
  var isDropdownDesktop = event.target.parentNode.classList.contains('p-dropdownitem');
  var isDropdownMobile = event.target.classList.contains('sub');

  if (!isDropdownDesktop && !isDropdownMobile) {
    return;
  }

  event.preventDefault();
  var countryCode = getCountries()[event.target.innerText];
  setCookie('country', countryCode, 30);
  window.location = buildUrlFromCountryCode(countryCode);
}

function buildUrlFromCountryCode(countryCode) {
  var rule = SUBDOMAINS_RULES.find(function (rule) {
    return rule.countries.includes(countryCode);
  });
  return rule.route || "//english.".concat(getHostname());
}

function getCountryFromCookie() {
  var countryCode = getCookie('country') || 'UK';
  var countries = getCountries();
  return Object.keys(countries).reduce(function (selectedCountry, code) {
    if (countries[code] === countryCode) {
      selectedCountry = code;
    }

    return selectedCountry;
  });
}

function validateLangDomain() {
  var countryCode = getCookie('country');
  var rules = SUBDOMAINS_RULES;

  if (!countryCode) {
    showBody();
    return;
  } // allow without subdomain


  if (['www', 'english'].includes(getSubdomain()) && ENGLISH_COUNTRIES.includes(countryCode)) {
    return showBody();
  }

  for (var i = 0; i < rules.length; i++) {
    var rule = rules[i];

    if (rule.countries.includes(countryCode)) {
      if (getSubdomain() === rule.subdomain) {
        showBody();
        return;
      }

      if (!rule.custom) {
        return window.location = rule.route + window.location.pathname;
      }
    }
  }

  showBody();
}

function selectCountryFromCookie() {
  document.querySelector('.p-navbarlink img + div').innerText = getCountryFromCookie();
  document.querySelector('.p-mobilemenulink .language').innerText = getCountryFromCookie();
}

function getCookie(name) {
  var v = document.cookie.match('(^|;) ?' + name + '=([^;]*)(;|$)');
  return v ? v[2] : null;
}

function setCookie(name, value, days) {
  var d = new Date();
  d.setTime(d.getTime() + 24 * 60 * 60 * 1000 * days);
  document.cookie = name + "=" + value + ";domain=" + getHostname() + ";path=/;expires=" + d.toGMTString();
}

function getHostname() {
  return window.location.host.split('.').slice(-2).join('.');
}

function getSubdomain() {
  return window.location.host.split('.')[0];
}

function showBody() {
  return document.body.style.display = 'block';
}

function getCountries() {
  return Array.from(document.querySelectorAll('*[data-country-code]')).reduce(function (hash, element) {
    hash[element.innerText.trim()] = element.getAttribute('data-country-code');
    return hash;
  }, {});
}

function getPathname() {
  return window.location.pathname.split('/')[1];
}

function showMarketplaceButton(data) {
  var isLoggedIn = getCookie('isLoggedIn');

  if (isLoggedIn === 'true') {
    document.querySelector('.sign-in').innerText = "Marketplace";
    document.querySelector('.register').style.display = 'none';
    document.querySelector('.mobilesignin').innerText = "Marketplace";
    document.querySelector('.mobileregister').style.display = 'none';
  }

  return;
}

function showTeachWithUsForm() {
  var params = new URLSearchParams(window.location.search);
  var data = params.get('data');
  data ? showTeachWithUs(JSON.parse(data)) : showContactUs();
}

function showContactUs() {
  return document.getElementById('contact-us').style.display = 'block';
}

function showTeachWithUs(data) {
  document.getElementById('firstName2').value = data.firstName;
  document.getElementById('lastName2').value = data.lastName;
  document.getElementById('email2').value = data.email;
  document.getElementById('typeContent2').value = data.typeContent;
  return document.getElementById('teach-us').style.display = 'block';
}

function showIframe() {
  document.querySelector('.iframe').src = window.location.href + '-' + getTermsOfUseSlug();
}

function getTermsOfUseSlug() {
  var lang = getSubdomain();

  switch (lang) {
    case 'es':
      return 'spanish';

    case 'pt':
      return 'portuguese';

    default:
      return 'english';
  }
}
