      // takes a selector, and adds the `data-hero-height-helper`
      // attribute for the `heroHeight` to be aware of
      function addHeroHelper(selector) {
        var elements = document.querySelectorAll(selector);
        for (var i = 0; i < elements.length; i++) {
          elements[i].setAttribute('data-hero-height-helper', '');
        }
      }

      // takes a selector, and removes the `data-hero-height-helper`
      // attribute for the `heroHeight` to be aware of
      function removeHeroHelper (selector) {
        var elements = document.querySelectorAll(selector);
        for (var i = 0; i < elements.length; i++) {
          elements[i].removeAttribute('data-hero-height-helper');
        }
      }

      // Looks for all `data-hero-height-helper` elements
      // and sets the `hero` height to 100vh minus the sum
      // of the hero helper heights
      function heroHeight () {
        var heroHeightHelpers = document.querySelectorAll('[data-hero-height-helper]');

        var helperHeight = 0;
        for (var i = 0; i < heroHeightHelpers.length; i++) {
          var currentHeight = heroHeightHelpers[i].getBoundingClientRect().height;
          helperHeight += currentHeight;
        }

        var heroes = document.querySelectorAll('.hero');
        for (var i = 0; i < heroes.length; i++) {
          if ('style' in heroes[i]) {
            var heightString = 'calc(100vh - ' + helperHeight + 'px)';
            heroes[i].style.height = heightString;
          }
        }
      }
