    <section>
     <h1>
      How to use?
     </h1>
     <div>
      <h3>
       <span class="command">
        MooTools Selectors
       </span>
      </h3>
      <p>
       Selectors for DOM Elements
      </p>
      <pre><code class="javascript"><span class="comment">// get elements by class</span>
$$(<span class="string">'.foo'</span>); <span class="comment">// or even: document.getElements('.foo');</span>

<span class="comment">// selector with different elements</span>
$$(<span class="string">'div.foo, div.bar, div.bar a'</span>);

<span class="comment">// get a single element</span>
document.getElement(<span class="string">'div.foo'</span>);</code></pre>
     </div>
     <div>
      <h3>
       <span class="command">
        Ajax!
       </span>
      </h3>
      <p>
       MooTools uses a Class called Request.
      </p>
      <pre><code class="javascript"><span class="comment">// create a new Class instance</span>
<span class="keyword">var</span> myRequest = <span class="keyword">new</span> Request({
    url: <span class="string">'getMyText.php'</span>,
    method: <span class="string">'get'</span>,
    onRequest: <span class="keyword">function</span>(){
        myElement.set(<span class="string">'text'</span>, <span class="string">'loading...'</span>);
    },
    onSuccess: <span class="keyword">function</span>(responseText){
        myElement.set(<span class="string">'text'</span>, responseText);
    },
    onFailure: <span class="keyword">function</span>(){
        myElement.set(<span class="string">'text'</span>, <span class="string">'Sorry, your request failed :('</span>);
    }
});

<span class="comment">// and to send it:</span>
myRequest.send(data);</code></pre>
      <p>
       To send a form it can be even more simple!
      </p>
      <pre><code class="javascript">myForm.send();
<span class="comment">// optionally you can add/change the form properties</span>
myForm.set(<span class="string">'send'</span>, {url: <span class="string">'contact.php'</span>, method: <span class="string">'get'</span>});</code></pre>
     </div>
     <div>
      <h3>
       <span class="command">
        How to create new DOM elements
       </span>
      </h3>
      <p>
       A simple MooTools Element example.
      </p>
      <pre><code class="javascript"><span class="comment">// the short way</span>
<span class="keyword">new</span> Element(<span class="string">'div#bar.foo'</span>);

<span class="comment">// using the element constructor</span>
<span class="keyword">new</span> Element(<span class="string">'div'</span>, {
    <span class="string">'class'</span>: <span class="string">'foo'</span>,
    id: <span class="string">'bar'</span>
});</code></pre>
     </div>
    </section>
    <section>
     <div>
      <h3>
       <span class="command">
        Event handlers
       </span>
      </h3>
      <p>
       Attach events to be updated on what's happening!
      </p>
      <pre><code class="javascript"><span class="comment">// attach a click event o a element</span>
myElement.addEvent(<span class="string">'click'</span>, <span class="keyword">function</span>(){
    alert(<span class="string">'clicked!'</span>);
});

<span class="comment">// attach several events at a time</span>
myElement.addEvents({
    mouseover: <span class="keyword">function</span>(){
        alert(<span class="string">'mouseover'</span>);
    },
    click: <span class="keyword">function</span>(){
        alert(<span class="string">'click'</span>);
    }
});</code></pre>
      <p>
       You can also remove, fire or clone events!
      </p>
      <pre><code class="javascript"><span class="comment">// remove a event</span>
myElement.removeEvent(type, fn);</code></pre>
     </div>
     <div>
      <h3>
       <span class="command">
        MooTools Class
       </span>
      </h3>
      <p>
       A simple MooTools Class example.
      </p>
      <pre><code class="javascript"><span class="keyword">var</span> Animal = <span class="keyword">new</span> Class({
    initialize: <span class="keyword">function</span>(age){
        <span class="keyword">this</span>.age = age;
    }
});

<span class="keyword">var</span> Cat = <span class="keyword">new</span> Class({
    Extends: Animal,
    initialize: <span class="keyword">function</span>(name, age){
        <span class="comment">// calls initalize method of Animal class</span>
        <span class="keyword">this</span>.parent(age);
        <span class="keyword">this</span>.name = name;
    }
});

<span class="keyword">var</span> myCat = <span class="keyword">new</span> Cat(<span class="string">'Micia'</span>, <span class="number">20</span>);
alert(myCat.name); <span class="comment">// alerts 'Micia'.</span>
alert(myCat.age); <span class="comment">// alerts 20.</span></code></pre>
     </div>
    </section>
