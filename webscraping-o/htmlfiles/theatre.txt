   var affiliateBannerId = 'affiliate-banner';

    document.body.classList.add(htmlWithoutBannerClass);

    if (!wasAffiliateBannerDismissed) {
      document.body.classList.remove(htmlWithoutBannerClass);
      document.body.classList.add(htmlBannerClass);

        addClickHandler();

    function updateBannerPosition () {
      var affiliateBanner = document.getElementById(affiliateBannerId);
      var header = document.querySelector('.js-topnav');

      if (affiliateBanner && header) {
        header.insertBefore(affiliateBanner.cloneNode(true), header.firstChild);

        affiliateBanner.style.display = 'none';
      }
    }

    function addClickHandler () {
      var closeButtons = Array.prototype.slice.call(document.querySelectorAll('.affiliate-banner-close-button'));

        closeButton.addEventListener('click', function () {
          document.body.classList.remove(htmlBannerClass);

            var affiliateBanners = Array.prototype.slice.call(document.querySelectorAll('.affiliate-banner'));

              affiliateBanner.parentNode.removeChild(affiliateBanner);
            });

            localStorage.setItem(bannerDismissedStorageKey, true);

            closeButton = null;
          }, 1000);
        });
      });
    }
