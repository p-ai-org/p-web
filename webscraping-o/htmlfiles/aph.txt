      jQuery(document).ready(function ($) {

        var carouselContainer = $('.js-slider-nav');

        var addPausePlay = function () {
        	// Add pause/play button
        	var navigationDots = $('.slick-dots');
        	if (navigationDots.length) {
        		var playPauseBtn = navigationDots.find('.play-pause-btn');
				if (!playPauseBtn.length) {
                    navigationDots.append('<li class="play-pause-container"><a href="#" class="play-pause-btn" aria-label="Pause"><em class="fa fa-fw fa-pause"></em></a></li>');

				    setTimeout(function () {
				        carouselContainer.slick('slickPlay');
				    }, 3000);
					
				}
        	}
        }
        
        $('.js-slider-nav').on('click', '.play-pause-btn', function (e) {
            e.preventDefault();
            if ($('.play-pause-btn em').hasClass('fa-pause')) {
                $('.play-pause-btn em').removeClass('fa-pause').addClass('fa-play');
                $('.play-pause-btn').attr('aria-label', 'Play');
                $('.js-slider-nav').slick('slickPause');
            }
            else {
                $('.play-pause-btn em').removeClass('fa-play').addClass('fa-pause');
                $('.play-pause-btn').attr('aria-label', 'Pause');
                $('.js-slider-nav').slick('slickPlay');
            }
        });

        if (carouselContainer.length) {
        	carouselContainer.on('breakpoint', function (event, slick) {
        		addPausePlay();
        	});

        	carouselContainer.not('.slick-initialized').slick({
                dots: true,
                slidesToShow: 5,
                centerMode: true,
                speed: 600,
                arrows: true,
                responsive: [{
                    breakpoint: 1024,
                    settings: {
                        slidesToShow: 2
                    }
                }, {
                    breakpoint: 480,
                    settings: {
                        slidesToShow: 1
                    }
                }]
            });

            addPausePlay();
        }
        $('.js-slider-nav').removeClass('hidden');
    });
